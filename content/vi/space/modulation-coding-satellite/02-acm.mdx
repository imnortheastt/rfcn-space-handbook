## Tai sao dieu nay quan trong

Mua co the lam suy giam tin hieu ve tinh bang Ka 10+ dB trong vai phut. Khong co ACM, nha khai thac phai thiet ke cho dieu kien xau nhat—lang phi 50%+ dung luong vao nhung ngay troi quang. Voi ACM, moi tram nhan duoc thong luong cao nhat ma lien ket hien tai cua no co the ho tro. Khi mua anh huong mot tram, chi tram do giam xuong MODCOD ben vung hon; moi nguoi khac van truyen hieu qua cao. ACM la ky thuat lam cho bang thong rong ve tinh hien dai kha thi ve kinh te.

## Ap dung

**Mo phong ACM va tinh toan loi ich:**

```python
import numpy as np

# Bang MODCOD voi nguong
MODCODS = [
    {"ten": "QPSK 1/2",   "hs": 0.99,  "nguong": 1.0},
    {"ten": "QPSK 2/3",   "hs": 1.32,  "nguong": 3.1},
    {"ten": "QPSK 3/4",   "hs": 1.49,  "nguong": 4.0},
    {"ten": "8PSK 2/3",   "hs": 1.98,  "nguong": 6.6},
    {"ten": "8PSK 3/4",   "hs": 2.23,  "nguong": 7.9},
    {"ten": "16APSK 2/3", "hs": 2.64,  "nguong": 9.0},
    {"ten": "16APSK 3/4", "hs": 2.97,  "nguong": 10.2},
    {"ten": "32APSK 3/4", "hs": 3.70,  "nguong": 12.2},
    {"ten": "32APSK 5/6", "hs": 4.12,  "nguong": 13.0},
]

def chon_modcod(snr_db, margin_db=1.0):
    """Chon MODCOD tot nhat cho SNR cho truoc voi margin."""
    co_san = snr_db - margin_db
    tot_nhat = None
    for mc in MODCODS:
        if mc["nguong"] <= co_san:
            if tot_nhat is None or mc["hs"] > tot_nhat["hs"]:
                tot_nhat = mc
    return tot_nhat if tot_nhat else MODCODS[0]

def mo_phong_su_kien_mua(snr_quang, suy_giam_toi_da, so_mau=100):
    """Mo phong su kien suy giam mua (suy giam chuong Gauss)."""
    t = np.linspace(0, 1, so_mau)
    suy_giam = suy_giam_toi_da * np.exp(-((t - 0.5) ** 2) / 0.02)
    return snr_quang - suy_giam

print("Mo phong ACM")
print("=" * 60)

# Kich ban: Lien ket Ka-band voi suy giam mua
snr_troi_quang = 15.0  # dB
suy_giam_mua_toi_da = 10.0  # dB
so_mau = 100

chuoi_snr = mo_phong_su_kien_mua(snr_troi_quang, suy_giam_mua_toi_da, so_mau)

print("\n1. SNR va MODCOD Trong Su kien Mua")
print("-" * 60)

# Theo doi thong ke
ccm_modcod = chon_modcod(snr_troi_quang - suy_giam_mua_toi_da)
acm_hieu_suat = []
ccm_hieu_suat = ccm_modcod["hs"]

# Cac diem hien thi
diem_hien_thi = [0, 25, 45, 50, 55, 75, 99]

print(f"  Thiet ke CCM: {ccm_modcod['ten']} (co dinh)")
print()
print(f"  {'Mau':<8} {'SNR (dB)':<10} {'ACM MODCOD':<14} {'HS ACM':<10} {'HS CCM':<10}")
print("-" * 60)

for i, snr in enumerate(chuoi_snr):
    acm_mc = chon_modcod(snr)
    acm_hieu_suat.append(acm_mc["hs"])
    
    if i in diem_hien_thi:
        print(f"  {i:<8} {snr:<10.1f} {acm_mc['ten']:<14} {acm_mc['hs']:<10.2f} {ccm_hieu_suat:<10.2f}")

# Tinh loi ich
hs_acm_tb = np.mean(acm_hieu_suat)
loi_ich_acm = (hs_acm_tb / ccm_hieu_suat - 1) * 100

print()
print("-" * 60)
print(f"  Hieu suat trung binh CCM: {ccm_hieu_suat:.2f} bps/Hz")
print(f"  Hieu suat trung binh ACM: {hs_acm_tb:.2f} bps/Hz")
print(f"  Loi ich ACM so voi CCM: +{loi_ich_acm:.1f}%")

# Mo phong tre hien
print("\n2. Anh huong cua Tre hien")
print("-" * 60)

def dem_thay_doi_modcod(chuoi_snr, tre_hien_db=0):
    """Dem cac chuyen doi MODCOD voi tre hien cho truoc."""
    thay_doi = 0
    mc_hien_tai = chon_modcod(chuoi_snr[0])
    
    for snr in chuoi_snr[1:]:
        mc_moi = chon_modcod(snr)
        
        if tre_hien_db > 0:
            if mc_moi["hs"] > mc_hien_tai["hs"]:
                if snr > mc_moi["nguong"] + tre_hien_db:
                    if mc_moi != mc_hien_tai:
                        thay_doi += 1
                        mc_hien_tai = mc_moi
            elif mc_moi["hs"] < mc_hien_tai["hs"]:
                if snr < mc_hien_tai["nguong"] - tre_hien_db:
                    if mc_moi != mc_hien_tai:
                        thay_doi += 1
                        mc_hien_tai = mc_moi
        else:
            if mc_moi != mc_hien_tai:
                thay_doi += 1
                mc_hien_tai = mc_moi
    
    return thay_doi

# Them nhieu de thuc te hon
snr_nhieu = chuoi_snr + np.random.normal(0, 0.5, len(chuoi_snr))

for hyst in [0, 0.5, 1.0, 2.0]:
    thay_doi = dem_thay_doi_modcod(snr_nhieu, hyst)
    print(f"  Tre hien {hyst:.1f} dB: {thay_doi} lan thay doi MODCOD")

# Kich ban nhieu tram
print("\n3. Kich ban ACM Nhieu Tram")
print("-" * 60)

tram = [
    {"ten": "Tram A", "snr_quang": 16, "phan_tram_suy_giam": 5},
    {"ten": "Tram B", "snr_quang": 14, "phan_tram_suy_giam": 10},
    {"ten": "Tram C", "snr_quang": 12, "phan_tram_suy_giam": 15},
    {"ten": "Tram D", "snr_quang": 10, "phan_tram_suy_giam": 20},
]

print(f"  {'Tram':<12} {'SNR Quang':<12} {'% Suy giam':<10} {'HS ACM':<10} {'HS CCM':<10}")
print("-" * 60)

tong_acm = 0
tong_ccm = 0

snr_xau_nhat = min(t["snr_quang"] * (1 - t["phan_tram_suy_giam"]/100*0.7) for t in tram)
ccm_mc = chon_modcod(snr_xau_nhat - 5)

for t in tram:
    mc_quang = chon_modcod(t["snr_quang"])
    snr_suy_giam = t["snr_quang"] - 5
    mc_suy_giam = chon_modcod(snr_suy_giam)
    
    phan_tram_quang = 100 - t["phan_tram_suy_giam"]
    hs_acm = (mc_quang["hs"] * phan_tram_quang + mc_suy_giam["hs"] * t["phan_tram_suy_giam"]) / 100
    
    tong_acm += hs_acm
    tong_ccm += ccm_mc["hs"]
    
    print(f"  {t['ten']:<12} {t['snr_quang']:<12} {t['phan_tram_suy_giam']:<10} {hs_acm:<10.2f} {ccm_mc['hs']:<10.2f}")

loi_ich_he_thong = (tong_acm / tong_ccm - 1) * 100
print()
print(f"  Loi ich ACM he thong: +{loi_ich_he_thong:.1f}%")

# Overhead kenh hoi
print("\n4. Yeu cau Kenh hoi ACM")
print("-" * 60)

khoang_bao_cao_ms = 500
kich_thuoc_bao_cao_bit = 32
so_tram = 1000

bao_cao_moi_giay = 1000 / khoang_bao_cao_ms
tong_bps = bao_cao_moi_giay * kich_thuoc_bao_cao_bit * so_tram

print(f"  Khoang bao cao: {khoang_bao_cao_ms} ms")
print(f"  Kich thuoc bao cao: {kich_thuoc_bao_cao_bit} bit")
print(f"  So tram: {so_tram}")
print(f"  Overhead kenh hoi: {tong_bps/1e6:.2f} Mbps")
print(f"  (Thuong <1% dung luong lien ket xuoi)")
```

<section class="lesson-pitfalls">

## Cam bay

1. **Bo qua do tre vong ACM.** RTT GEO la 500ms. Suy giam nhanh (scintillation) co the nhanh hon ACM co the phan hoi.

2. **Khong co tre hien.** Khong co margin, MODCOD dao dong nhanh, gay mat thong luong do decoder reset.

3. **Bao cao SNR cu.** Neu kenh hoi tac, du lieu SNR cu gay chon sai MODCOD.

4. **Bat doi xung len/xuong.** Mua suy giam ca hai huong. ACM len (tu tram) can dieu khien rieng.

5. **ACM theo tram vs theo chum.** HTS voi chum diem hep co the dung ACM theo chum (tat ca tram trong chum dung chung MODCOD).

6. **Gia dinh ACM sua tat ca margin.** Ton that tinh (lech chi huong anten, thiet bi lao hoa) van can margin.
