## Tai sao dieu nay quan trong

Moi lien ket ve tinh deu doi mat voi rang buoc co ban giong nhau: ban co the co thong luong cao HOAC hoat dong ben trong dieu kien xau — khong the ca hai dong thoi. Danh doi nay chi phoi moi quyet dinh thiet ke tu kich thuoc anten tram den phan bo margin mua. Hieu danh doi hieu suat pho cho phep ban toi uu dung luong ma khong hy sinh tinh san sang, va giai thich tai sao cac dich vu khac nhau (truyen hinh quang ba vs bang rong hang hai) dua ra lua chon thiet ke hoan toan khac nhau.

## Ap dung

**Phan tich va toi uu hieu suat pho:**

```python
import numpy as np

def gioi_han_shannon(snr_db):
    """Gioi han Shannon tinh bang bps/Hz cho SNR cho truoc."""
    snr_tuyen_tinh = 10 ** (snr_db / 10)
    return np.log2(1 + snr_tuyen_tinh)

def snr_cho_hieu_suat(eta):
    """SNR yeu cau (dB) cho hieu suat pho tai gioi han Shannon."""
    snr_tuyen_tinh = 2 ** eta - 1
    return 10 * np.log10(snr_tuyen_tinh)

# Diem tham chieu DVB-S2
MODCODS = [
    {"ten": "QPSK 1/4",   "hs": 0.49, "nguong": -2.35},
    {"ten": "QPSK 1/2",   "hs": 0.99, "nguong":  1.00},
    {"ten": "QPSK 3/4",   "hs": 1.49, "nguong":  4.03},
    {"ten": "8PSK 2/3",   "hs": 1.98, "nguong":  6.62},
    {"ten": "16APSK 3/4", "hs": 2.97, "nguong": 10.21},
    {"ten": "32APSK 4/5", "hs": 3.95, "nguong": 13.05},
    {"ten": "32APSK 9/10","hs": 4.45, "nguong": 15.07},
]

print("Phan tich Danh doi Hieu suat Pho")
print("=" * 70)

print("\n1. Khoang cach den Gioi han Shannon theo MODCOD")
print("-" * 70)
print(f"  {'MODCOD':<14} {'Hieu suat':<12} {'Nguong':<12} {'Shannon':<12} {'Gap (dB)':<10}")
print("-" * 70)

for mc in MODCODS:
    shannon_snr = snr_cho_hieu_suat(mc["hs"])
    gap = mc["nguong"] - shannon_snr
    print(f"  {mc['ten']:<14} {mc['hs']:<12.2f} {mc['nguong']:<12.2f} {shannon_snr:<12.2f} {gap:<10.2f}")

# Phan tich phan bo margin
print("\n2. Chien luoc Phan bo Margin (C/N co san 14 dB)")
print("-" * 70)

cn_co_san = 14.0
ton_that_trien_khai = 0.5

chien_luoc = [
    {"ten": "Thong luong toi da", "margin_mua": 0.0, "san_sang": "~95%"},
    {"ten": "Can bang", "margin_mua": 3.0, "san_sang": "99.5%"},
    {"ten": "San sang cao", "margin_mua": 6.0, "san_sang": "99.9%"},
    {"ten": "Cap quang ba", "margin_mua": 9.0, "san_sang": "99.99%"},
]

for cl in chien_luoc:
    snr_dung_duoc = cn_co_san - ton_that_trien_khai - cl["margin_mua"]
    
    mc_tot_nhat = None
    for mc in MODCODS:
        if mc["nguong"] <= snr_dung_duoc - 0.5:
            mc_tot_nhat = mc
    
    if mc_tot_nhat:
        print(f"  {cl['ten']}:")
        print(f"    Margin mua: {cl['margin_mua']:.1f} dB → san sang {cl['san_sang']}")
        print(f"    SNR dung duoc: {snr_dung_duoc:.1f} dB")
        print(f"    MODCOD tot nhat: {mc_tot_nhat['ten']} → {mc_tot_nhat['hs']:.2f} bps/Hz")
        print()

# Tinh loi dung luong ACM
print("3. Phan tich Dung luong ACM vs CCM")
print("-" * 70)

phan_bo_fade = [
    {"phan_tram": 70, "fade_db": 0.0, "dieu_kien": "Troi quang"},
    {"phan_tram": 20, "fade_db": 2.0, "dieu_kien": "Fade nhe"},
    {"phan_tram":  8, "fade_db": 5.0, "dieu_kien": "Vua phai"},
    {"phan_tram":  1.5, "fade_db": 8.0, "dieu_kien": "Nang"},
    {"phan_tram":  0.5, "fade_db": 12.0, "dieu_kien": "Nghiem trong"},
]

cn_troi_quang = 14.0
toc_do_ky_hieu = 30.0

def chon_modcod(snr_db, margin=0.5):
    tot_nhat = None
    for mc in MODCODS:
        if mc["nguong"] <= snr_db - margin:
            tot_nhat = mc
    return tot_nhat

print(f"  C/N troi quang: {cn_troi_quang:.1f} dB")
print(f"  Toc do ky hieu: {toc_do_ky_hieu:.0f} Msps")
print()

print("  Van hanh ACM:")
print(f"    {'TG %':<8} {'Fade':<8} {'C/N HD':<10} {'MODCOD':<14} {'Hieu suat':<12}")
print("-" * 70)

hs_co_trong = 0.0
for fade in phan_bo_fade:
    cn_hieu_dung = cn_troi_quang - fade["fade_db"]
    mc = chon_modcod(cn_hieu_dung)
    if mc:
        hs = mc["hs"]
        ten = mc["ten"]
    else:
        hs = 0.0
        ten = "MAT SONG"
    
    hs_co_trong += (fade["phan_tram"] / 100) * hs
    print(f"    {fade['phan_tram']:<8.1f} {fade['fade_db']:<8.1f} {cn_hieu_dung:<10.1f} {ten:<14} {hs:<12.2f}")

toc_do_acm = toc_do_ky_hieu * hs_co_trong
print(f"\n  Hieu suat co trong ACM: {hs_co_trong:.2f} bps/Hz")
print(f"  Toc do du lieu TB ACM: {toc_do_acm:.1f} Mbps")

# Baseline CCM
modcod_ccm = chon_modcod(cn_troi_quang - 3.0)
toc_do_ccm = toc_do_ky_hieu * modcod_ccm["hs"]
print(f"\n  CCM co dinh ({modcod_ccm['ten']}): {modcod_ccm['hs']:.2f} bps/Hz")
print(f"  Toc do du lieu CCM: {toc_do_ccm:.1f} Mbps")
print(f"\n  Loi dung luong ACM: {toc_do_acm / toc_do_ccm:.2f}x ({(toc_do_acm / toc_do_ccm - 1) * 100:.0f}% hon)")

# Du lieu truc quan danh doi
print("\n4. Danh doi Hieu suat vs San sang")
print("-" * 70)

margins = [0, 2, 4, 6, 8, 10]
print(f"  {'Margin Mua':<14} {'SNR Dung':<12} {'MODCOD TN':<14} {'Hieu suat':<12}")
print("-" * 70)

for margin in margins:
    dung_duoc = cn_troi_quang - ton_that_trien_khai - margin
    mc = chon_modcod(dung_duoc)
    if mc:
        print(f"  {margin:<14} {dung_duoc:<12.1f} {mc['ten']:<14} {mc['hs']:<12.2f}")
    else:
        print(f"  {margin:<14} {dung_duoc:<12.1f} {'(khong hop le)':<14} {'0.00':<12}")

# Phan tich chi phi bien hieu suat
print("\n5. Chi phi Bien cua Hieu suat")
print("-" * 70)
print("  Moi bps/Hz them vao ton NHIEU SNR hon:")
print()

hs_truoc = 0
snr_truoc = -3
for mc in MODCODS:
    delta_hs = mc["hs"] - hs_truoc
    delta_snr = mc["nguong"] - snr_truoc
    chi_phi = delta_snr / delta_hs if delta_hs > 0 else 0
    print(f"  {hs_truoc:.2f} → {mc['hs']:.2f} bps/Hz: ton {delta_snr:.1f} dB ({chi_phi:.1f} dB moi bps/Hz)")
    hs_truoc = mc["hs"]
    snr_truoc = mc["nguong"]
```

**Quy tac danh doi chinh:**

| Quy tac | Ham y |
|---------|-------|
| Gioi han Shannon la cung | Khong MODCOD nao vuot log₂(1+SNR) |
| Loi ich giam dan | MODCOD cao ton nhieu dB hon moi bps/Hz |
| ACM phuc hoi margin | 30-70% loi dung luong so voi CCM |
| Margin mua dat | 3 dB margin = mat 50% co hoi dung luong |
| Kich thuoc tram doi lay margin | Dia 6 dB lon hon → cung dung luong, san sang 4× |

<section class="lesson-pitfalls">

## Cam bay

1. **Thiet ke cho dung luong dinh.** Lien ket 5 bps/Hz that bai 10% thoi gian te hon 2.5 bps/Hz o 99.9%.

2. **Bo qua dong hoc suy giam.** Loi dung luong ACM gia dinh thong ke fade. Mat song tuong quan (bao) giam loi thuc te.

3. **Quen margin trien khai.** He thong thuc te mat 0.5-1.5 dB cho may thu khong hoan hao, nhieu pha, phuc hoi dinh thi.

4. **Gia dinh margin doi xung.** Mua anh huong downlink nhieu hon uplink o Ku/Ka. Lap ngan sach bat doi xung.

5. **Coi Shannon la co the dat.** He thong thuc te cach 0.5-1.5 dB. Dung lap ke hoach cho dieu bat kha thi.

6. **Toi uu qua cho mot chi so.** San sang × dung luong thuong la muc tieu dung, khong phai rieng le.
