## Tai sao dieu nay quan trong

Moi vat the duoc theo doi tren quy dao — tu ISS den mot con oc bi bo di — deu co Bo Phan tu Hai dong (TLE) duoc cong bo boi Space-Track hoac CelesTrak. TLE la ngon ngu chung cua theo doi ve tinh: phan mem tram mat dat, he thong tranh va cham, va nguoi theo doi nghiep du deu su dung dinh dang nay. Neu ban muon du doan vi tri ve tinh, ban bat dau voi TLE.

## Ap dung

**Trinh phan tich va xac thuc TLE:**

```python
import re
from datetime import datetime, timedelta

def phan_tich_tle(dong1, dong2, ten=None):
    """
    Phan tich TLE thanh du lieu co cau truc.
    """
    if len(dong1) != 69 or len(dong2) != 69:
        raise ValueError(f"Dong TLE phai la 69 ky tu. Nhan {len(dong1)}, {len(dong2)}")
    
    if dong1[0] != '1' or dong2[0] != '2':
        raise ValueError("Dong 1 phai bat dau bang '1', Dong 2 bang '2'")
    
    norad_id_1 = int(dong1[2:7])
    norad_id_2 = int(dong2[2:7])
    if norad_id_1 != norad_id_2:
        raise ValueError(f"NORAD ID khong khop: {norad_id_1} vs {norad_id_2}")
    
    phan_loai = dong1[7]
    ky_hieu_qt = dong1[9:17].strip()
    
    nam_epoch = int(dong1[18:20])
    ngay_epoch = float(dong1[20:32])
    
    if nam_epoch < 57:
        nam_epoch += 2000
    else:
        nam_epoch += 1900
    
    epoch = datetime(nam_epoch, 1, 1) + timedelta(days=ngay_epoch - 1)
    
    ndot = float(dong1[33:43])
    
    bstar_str = dong1[53:61].strip()
    if bstar_str:
        dau = -1 if bstar_str[0] == '-' else 1
        bstar_str = bstar_str.lstrip('-').lstrip('+').lstrip()
        phan_dinh = float("0." + bstar_str[:-2].replace(" ", "0"))
        so_mu = int(bstar_str[-2:])
        bstar = dau * phan_dinh * (10 ** so_mu)
    else:
        bstar = 0.0
    
    do_nghieng = float(dong2[8:16])
    raan = float(dong2[17:25])
    do_lech_tam = float("0." + dong2[26:33])
    goc_can_diem = float(dong2[34:42])
    di_so_tb = float(dong2[43:51])
    chuyen_dong_tb = float(dong2[52:63])
    so_vong = int(dong2[63:68])
    
    return {
        "ten": ten,
        "norad_id": norad_id_1,
        "phan_loai": phan_loai,
        "ky_hieu_qt": ky_hieu_qt,
        "epoch": epoch,
        "nam_epoch": nam_epoch,
        "ngay_epoch": ngay_epoch,
        "ndot": ndot,
        "bstar": bstar,
        "do_nghieng_do": do_nghieng,
        "raan_do": raan,
        "do_lech_tam": do_lech_tam,
        "goc_can_diem_do": goc_can_diem,
        "di_so_tb_do": di_so_tb,
        "chuyen_dong_tb_vong_ngay": chuyen_dong_tb,
        "so_vong": so_vong
    }

def tinh_kiem_tra_tong(dong):
    """Tinh kiem tra tong TLE cho mot dong."""
    tong = 0
    for c in dong[:-1]:
        if c.isdigit():
            tong += int(c)
        elif c == '-':
            tong += 1
    return tong % 10

def xac_thuc_kiem_tra_tong(dong1, dong2):
    """Xac thuc kiem tra tong TLE."""
    kiem_tra1 = tinh_kiem_tra_tong(dong1)
    kiem_tra2 = tinh_kiem_tra_tong(dong2)
    
    ky_vong1 = int(dong1[-1])
    ky_vong2 = int(dong2[-1])
    
    return {
        "dong1_hop_le": kiem_tra1 == ky_vong1,
        "dong2_hop_le": kiem_tra2 == ky_vong2,
        "dong1_tinh_duoc": kiem_tra1,
        "dong1_ky_vong": ky_vong1,
        "dong2_tinh_duoc": kiem_tra2,
        "dong2_ky_vong": ky_vong2
    }

def suy_ra_tham_so_quy_dao(tle):
    """Suy ra cac tham so quy dao bo sung tu TLE."""
    import math
    
    mu = 398600.4418
    
    n_rad_giay = tle["chuyen_dong_tb_vong_ngay"] * 2 * math.pi / 86400
    
    a = (mu / (n_rad_giay ** 2)) ** (1/3)
    
    e = tle["do_lech_tam"]
    do_cao_can_diem = a * (1 - e) - 6378.137
    do_cao_vien_diem = a * (1 + e) - 6378.137
    
    chu_ky_phut = 1440 / tle["chuyen_dong_tb_vong_ngay"]
    
    return {
        "ban_truc_lon_km": a,
        "do_cao_can_diem_km": do_cao_can_diem,
        "do_cao_vien_diem_km": do_cao_vien_diem,
        "chu_ky_quy_dao_phut": chu_ky_phut
    }

print("Demo Trinh Phan tich TLE")
print("=" * 70)

iss_tle = """ISS (ZARYA)
1 25544U 98067A   24001.50000000  .00016717  00000-0  10270-3 0  9993
2 25544  51.6400 247.4627 0006703 130.5360 325.0288 15.72125391463537"""

cac_dong = iss_tle.strip().split('\n')
ten = cac_dong[0] if len(cac_dong) == 3 else None
dong1 = cac_dong[1] if len(cac_dong) == 3 else cac_dong[0]
dong2 = cac_dong[2] if len(cac_dong) == 3 else cac_dong[1]

print("\n1. TLE Goc")
print("-" * 70)
if ten:
    print(f"  {ten}")
print(f"  {dong1}")
print(f"  {dong2}")

print("\n2. Xac thuc Kiem tra tong")
print("-" * 70)
kiem_tra = xac_thuc_kiem_tra_tong(dong1, dong2)
print(f"  Dong 1: tinh={kiem_tra['dong1_tinh_duoc']}, ky_vong={kiem_tra['dong1_ky_vong']} → {'✓' if kiem_tra['dong1_hop_le'] else '✗'}")
print(f"  Dong 2: tinh={kiem_tra['dong2_tinh_duoc']}, ky_vong={kiem_tra['dong2_ky_vong']} → {'✓' if kiem_tra['dong2_hop_le'] else '✗'}")

print("\n3. Cac Truong da Phan tich")
print("-" * 70)
tle = phan_tich_tle(dong1, dong2, ten)

print(f"""
  Ve tinh:          {tle['ten']} (NORAD {tle['norad_id']})
  Ky hieu Quoc te:  {tle['ky_hieu_qt']}
  
  Epoch:            {tle['epoch'].strftime('%Y-%m-%d %H:%M:%S')} UTC
  
  Cac Phan tu Quy dao:
    Do nghieng:       {tle['do_nghieng_do']:.4f}°
    RAAN:             {tle['raan_do']:.4f}°
    Do lech tam:      {tle['do_lech_tam']:.7f}
    Goc Can diem:     {tle['goc_can_diem_do']:.4f}°
    Di so Trung binh: {tle['di_so_tb_do']:.4f}°
    Chuyen dong TB:   {tle['chuyen_dong_tb_vong_ngay']:.8f} vong/ngay
""")

print("4. Tham so Quy dao Suy ra")
print("-" * 70)
suy_ra = suy_ra_tham_so_quy_dao(tle)
print(f"""
  Ban truc lon:       {suy_ra['ban_truc_lon_km']:.2f} km
  Do cao can diem:    {suy_ra['do_cao_can_diem_km']:.2f} km
  Do cao vien diem:   {suy_ra['do_cao_vien_diem_km']:.2f} km
  Chu ky quy dao:     {suy_ra['chu_ky_quy_dao_phut']:.2f} phut
""")
```

**Cong thuc chinh:**

| Tham so | Cong thuc |
|---------|-----------|
| Ban truc lon | a = (μ / n²)^(1/3) voi n tinh bang rad/s |
| Do cao can diem | a × (1 - e) - R_trai_dat |
| Do cao vien diem | a × (1 + e) - R_trai_dat |
| Chu ky quy dao | 2π / n |

<section class="lesson-pitfalls">

## Cam bay

1. **Su dung TLE cu.** TLE giam ~1 km/ngay cho LEO. Luon su dung TLE moi nhat tu Space-Track hoac CelesTrak.

2. **Giai thich nam sai.** 00-56 → 2000-2056; 57-99 → 1957-1999. Lam sai se lech epoch 100 nam.

3. **Quen thap phan ngam dinh.** Do lech tam "0006703" la 0.0006703, khong phai 6703. BSTAR "10270-3" la 0.00010270.

4. **Nham lan RAAN va goc can diem.** RAAN do tu diem xuan phan den diem len. Goc can diem do tu diem len den can diem.

5. **Bo qua kiem tra tong.** TLE hong tao ra du doan sai lech lon. Luon xac thuc kiem tra tong truoc khi su dung.
