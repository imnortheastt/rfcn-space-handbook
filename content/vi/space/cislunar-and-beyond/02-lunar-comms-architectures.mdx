## Tai sao dieu nay quan trong

Cuc nam Mat Trang—muc tieu cua Artemis—khong bao gio nhin thay truc tiep tu Trai dat. Khong co ha tang chuyen tiep, phi hanh gia o do se khong co lien lac voi trung tam dieu khien trong cac EVA quan trong. Nhieu quoc gia dang thiet ke kien truc thong tin Mat Trang: LunaNet cua NASA, Moonlight cua ESA, va cac nha cung cap thuong mai. Hieu cac he thong nay cho thay mang cislunar khac voi chom sao ve tinh lay Trai dat lam trung tam nhu the nao va tai sao ve tinh chuyen tiep la thiet yeu cho su hien dien Mat Trang ben vung.

## Ap dung

**Phan tich phu song chuyen tiep Mat Trang:**

```python
import numpy as np

def fspl_db(khoang_cach_km, tan_so_ghz):
    """Ton hao Duong truyen Khong gian Tu do tinh dB."""
    return 20 * np.log10(khoang_cach_km) + 20 * np.log10(tan_so_ghz) + 92.45

def link_cn0(eirp_dbw, khoang_cach_km, tan_so_ghz, g_tren_t_dbk, ton_hao_db=0):
    """Tinh C/N0 cho lien ket."""
    k_dbw = -228.6
    fspl = fspl_db(khoang_cach_km, tan_so_ghz)
    return eirp_dbw - fspl - ton_hao_db + g_tren_t_dbk - k_dbw

def toc_do_tu_cn0(cn0_dbhz, ebn0_can_db):
    """Tinh toc do du lieu dat duoc tu C/N0."""
    return 10 ** ((cn0_dbhz - ebn0_can_db) / 10)

# Tham so nguoi dung be mat
print("Phan tich Thong tin Be mat Mat Trang")
print("=" * 60)

# Dinh nghia cac loai nguoi dung
nguoi_dung = {
    "Phi hanh gia EVA": {"tx_w": 2, "do_loi_dbi": 0, "tan_so_ghz": 2.0},
    "Xe tu hanh": {"tx_w": 20, "do_loi_dbi": 6, "tan_so_ghz": 8.0},
    "Tau ha canh": {"tx_w": 50, "do_loi_dbi": 30, "tan_so_ghz": 26.0},
    "Can cu co dinh": {"tx_w": 100, "do_loi_dbi": 40, "tan_so_ghz": 32.0},
}

# Tham so ve tinh chuyen tiep
chuyen_tiep_gt = 10  # dB/K (chuyen tiep S-band)
chuyen_tiep_gt_ka = 20  # dB/K (chuyen tiep Ka-band)

# Khoang cach theo loai quy dao
quy_dao_chuyen_tiep = {
    "LLO (300 km)": [300, 1500],  # [min, max] km
    "ELFO can diem": [700, 2000],
    "ELFO vien diem": [6000, 10000],
    "Gateway": [3000, 70000],
}

print("\n1. EIRP Nguoi dung Be mat")
print("-" * 60)

print(f"  {'Nguoi dung':<20} {'TX(W)':<8} {'DL(dBi)':<10} {'EIRP(dBW)':<10}")
print("-" * 60)

for ten, tham_so in nguoi_dung.items():
    tx_dbw = 10 * np.log10(tham_so["tx_w"])
    eirp = tx_dbw + tham_so["do_loi_dbi"]
    print(f"  {ten:<20} {tham_so['tx_w']:<8} {tham_so['do_loi_dbi']:<10} {eirp:.1f}")

# Ngan sach lien ket cho moi to hop
print("\n2. C/N₀ Be mat-Chuyen tiep (dBHz)")
print("-" * 60)

# Dung phi hanh gia lam demo
phg = nguoi_dung["Phi hanh gia EVA"]
phg_eirp = 10 * np.log10(phg["tx_w"]) + phg["do_loi_dbi"]

print(f"  Phi hanh gia EVA (EIRP = {phg_eirp:.1f} dBW, {phg['tan_so_ghz']} GHz)")
print()
print(f"  {'Quy dao CT':<18} {'K.cach Min':<12} {'K.cach Max':<12}")
print("-" * 60)

ebn0_giong_noi = 6  # dB cho giong noi

for qd, (d_min, d_max) in quy_dao_chuyen_tiep.items():
    cn0_max = link_cn0(phg_eirp, d_min, phg["tan_so_ghz"], chuyen_tiep_gt)
    cn0_min = link_cn0(phg_eirp, d_max, phg["tan_so_ghz"], chuyen_tiep_gt)
    rate_max = toc_do_tu_cn0(cn0_max, ebn0_giong_noi) / 1000  # kbps
    rate_min = toc_do_tu_cn0(cn0_min, ebn0_giong_noi) / 1000
    print(f"  {qd:<18} {cn0_max:.1f} dBHz    {cn0_min:.1f} dBHz")
    print(f"  {'':18} ({rate_max:.0f} kbps)   ({rate_min:.0f} kbps)")

# Lien ket toc do cao tau ha canh
print("\n3. Lien ket Toc do cao Tau ha canh (Ka-band)")
print("-" * 60)

thc = nguoi_dung["Tau ha canh"]
thc_eirp = 10 * np.log10(thc["tx_w"]) + thc["do_loi_dbi"]

print(f"  Tau ha canh: EIRP = {thc_eirp:.1f} dBW @ {thc['tan_so_ghz']} GHz")
print()

ebn0_du_lieu = 3  # dB cho LDPC toc do cao

for qd, (d_min, d_max) in quy_dao_chuyen_tiep.items():
    cn0_max = link_cn0(thc_eirp, d_min, thc["tan_so_ghz"], chuyen_tiep_gt_ka)
    cn0_min = link_cn0(thc_eirp, d_max, thc["tan_so_ghz"], chuyen_tiep_gt_ka)
    rate_max = toc_do_tu_cn0(cn0_max, ebn0_du_lieu) / 1e6  # Mbps
    rate_min = toc_do_tu_cn0(cn0_min, ebn0_du_lieu) / 1e6
    print(f"  {qd:<18} {rate_max:.1f} Mbps    {rate_min:.2f} Mbps")

# So sanh phu song
print("\n4. Phu song Cuc Nam theo Kien truc")
print("-" * 60)

phu_song = [
    ("Truc tiep den T.Dat", "0-50%", "Chi khi vien ho nhin thay"),
    ("1 ve tinh ELFO", "~60%", "Nua cua quy dao 12 gio"),
    ("2 ve tinh ELFO", "~90%", "Phu song chong lan"),
    ("4 ve tinh ELFO", "~99%", "Du phong day du"),
    ("Chom sao LLO", "~95-99%", "Can 6-12 ve tinh"),
    ("Chi Gateway", "~20-30%", "Tam nhin NRHO han che"),
]

print(f"  {'Kien truc':<20} {'Phu song':<10} {'Ghi chu':<30}")
print("-" * 60)

for ktruc, ps, gc in phu_song:
    print(f"  {ktruc:<20} {ps:<10} {gc:<30}")

# So sanh voi GEO Trai dat
print("\n5. So sanh: Be mat Mat Trang vs Di dong Trai dat")
print("-" * 60)

so_sanh = [
    ("", "Di dong T.Dat", "EVA Mat Trang"),
    ("Khoang cach", "36,000 km (GEO)", "1,000 km (ELFO)"),
    ("Ton hao dt @ 2 GHz", "189 dB", "159 dB"),
    ("Loi the", "-", "De hon 30 dB!"),
    ("EIRP nguoi dung", "0.25 W, 0 dBi", "2 W, 0 dBi"),
    ("Toc do dat duoc", "~100 kbps", "~100-500 kbps"),
]

for muc, trai_dat, mat_trang in so_sanh:
    print(f"  {muc:<18} {trai_dat:<20} {mat_trang:<20}")
```

## Cam bay

1. **Gia dinh tam nhin truc tiep Trai dat.** Day ho cuc nam thuong khong co duong nhin Trai dat.

2. **Coi Gateway la chuyen tiep chinh.** Tam nhin NRHO den cuc nam chi 20-30%. Gateway la du phong, khong phai chinh.

3. **Danh gia thap rang buoc EVA.** Thiet bi phi hanh gia la 2 W da huong—khong the dong lien ket toc do cao.

4. **Bo qua che khuat dia hinh Mat Trang.** Nui va vach ho chan tin hieu ngay ca khi chuyen tiep o tren dau.

5. **Thiet ke diem hong don.** An toan phi hanh gia can duong thong tin du phong.

6. **Quen moi truong nhiet.** Chu ky ngay/dem Mat Trang anh huong chi huong anten va dien tu.
