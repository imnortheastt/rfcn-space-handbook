## Tai sao dieu nay quan trong

Trai voi truc giac, them lan dot thu ba doi khi co the su dung it nhien lieu hon chuyen Hohmann "toi uu". Chuyen bi-elliptic khai thac hieu ung Oberth—dot khi ban dang di nhanh nhat cho nhieu thay doi nang luong hon moi don vi nhien lieu. Cho thay doi quy dao voi ti le ban kinh lon hon 11.94, bi-elliptic thang ve delta-v. Danh doi: thoi gian chuyen dai hon nhieu.

## Ap dung

**Tinh toan chuyen bi-elliptic:**

```python
import numpy as np

MU_TRAI_DAT = 398600.4418  # km³/s²

def hohmann_delta_v(r1, r2, mu=MU_TRAI_DAT):
    """Tinh delta-v chuyen Hohmann."""
    a_t = (r1 + r2) / 2
    
    v1_tron = np.sqrt(mu / r1)
    v1_chuyen = np.sqrt(mu * (2/r1 - 1/a_t))
    delta_v1 = v1_chuyen - v1_tron
    
    v2_tron = np.sqrt(mu / r2)
    v2_chuyen = np.sqrt(mu * (2/r2 - 1/a_t))
    delta_v2 = v2_tron - v2_chuyen
    
    return abs(delta_v1) + abs(delta_v2)

def bielliptic_delta_v(r1, r2, rb, mu=MU_TRAI_DAT):
    """
    Tinh delta-v chuyen bi-elliptic.
    """
    # Ban truc lon
    a1 = (r1 + rb) / 2
    a2 = (r2 + rb) / 2
    
    # Dot 1: tai r1, tang toc sang ellipse 1
    v1_tron = np.sqrt(mu / r1)
    v1_chuyen1 = np.sqrt(mu * (2/r1 - 1/a1))
    delta_v1 = abs(v1_chuyen1 - v1_tron)
    
    # Dot 2: tai rb, chuyen sang ellipse 2
    v_rb_chuyen1 = np.sqrt(mu * (2/rb - 1/a1))
    v_rb_chuyen2 = np.sqrt(mu * (2/rb - 1/a2))
    delta_v2 = abs(v_rb_chuyen2 - v_rb_chuyen1)
    
    # Dot 3: tai r2, tron hoa
    v2_chuyen2 = np.sqrt(mu * (2/r2 - 1/a2))
    v2_tron = np.sqrt(mu / r2)
    delta_v3 = abs(v2_tron - v2_chuyen2)
    
    return delta_v1, delta_v2, delta_v3, delta_v1 + delta_v2 + delta_v3

def thoi_gian_bielliptic(r1, r2, rb, mu=MU_TRAI_DAT):
    """Tinh thoi gian chuyen bi-elliptic."""
    a1 = (r1 + rb) / 2
    a2 = (r2 + rb) / 2
    
    t1_nua = np.pi * np.sqrt(a1**3 / mu)
    t2_nua = np.pi * np.sqrt(a2**3 / mu)
    
    return (t1_nua + t2_nua) / 3600  # gio

def tim_rb_toi_uu(r1, r2, mu=MU_TRAI_DAT, pham_vi_rb=None):
    """Tim rb toi uu giam thieu tong delta-v."""
    if pham_vi_rb is None:
        pham_vi_rb = np.linspace(r2 * 1.1, r2 * 10, 100)
    
    rb_tot_nhat = pham_vi_rb[0]
    dv_tot_nhat = float('inf')
    
    for rb in pham_vi_rb:
        _, _, _, tong = bielliptic_delta_v(r1, r2, rb, mu)
        if tong < dv_tot_nhat:
            dv_tot_nhat = tong
            rb_tot_nhat = rb
    
    return rb_tot_nhat, dv_tot_nhat

# Tinh toan vi du
print("Phan tich Chuyen Bi-elliptic")
print("=" * 60)

# So sanh voi Hohmann cho cac ti le ban kinh khac nhau
print("\n1. Bi-elliptic vs Hohmann theo Ti le Ban kinh")
print("-" * 60)

r1 = 6771  # Ban kinh LEO (km)
print(f"  Quy dao dau: {r1} km ban kinh")
print(f"")
print(f"  {'Ti le R':>8} {'r₂ (km)':>12} {'Hohmann':>10} {'Bi-elliptic':>12} {'Thang':>10}")
print(f"  {'':>8} {'':>12} {'(km/s)':>10} {'(km/s)':>12} {'':>10}")
print("-" * 60)

for ti_le in [5, 10, 11.94, 15, 20, 50]:
    r2 = r1 * ti_le
    dv_hoh = hohmann_delta_v(r1, r2)
    
    rb_opt, dv_bie = tim_rb_toi_uu(r1, r2)
    
    thang = "Hohmann" if dv_hoh <= dv_bie else "Bi-elliptic"
    print(f"  {ti_le:>8.2f} {r2:>12.0f} {dv_hoh:>10.3f} {dv_bie:>12.3f} {thang:>10}")

# Phan tich chi tiet cho truong hop cu the
print("\n2. Phan tich Chi tiet Bi-elliptic (R = 50)")
print("-" * 60)
r1 = 6771
r2 = r1 * 50
gia_tri_rb = [r2 * 1.5, r2 * 2, r2 * 5, r2 * 10]

print(f"  r₁ = {r1} km, r₂ = {r2} km")
print(f"  Hohmann Δv = {hohmann_delta_v(r1, r2):.3f} km/s")
print(f"")
print(f"  {'r_b (km)':>15} {'Δv₁':>8} {'Δv₂':>8} {'Δv₃':>8} {'Tong':>8} {'TG':>10}")
print(f"  {'':>15} {'(km/s)':>8} {'(km/s)':>8} {'(km/s)':>8} {'(km/s)':>8} {'(gio)':>10}")

for rb in gia_tri_rb:
    dv1, dv2, dv3, tong = bielliptic_delta_v(r1, r2, rb)
    thoi_gian_h = thoi_gian_bielliptic(r1, r2, rb)
    print(f"  {rb:>15.0f} {dv1:>8.3f} {dv2:>8.3f} {dv3:>8.3f} {tong:>8.3f} {thoi_gian_h:>10.1f}")

# Truong hop thuc te: LEO sang GEO
print("\n3. Truong hop Thuc te: LEO (400 km) sang GEO")
print("-" * 60)
r_leo = 6771
r_geo = 42157
ti_le = r_geo / r_leo

print(f"  Ti le R = {ti_le:.2f}")
print(f"")
dv_hoh = hohmann_delta_v(r_leo, r_geo)
print(f"  Hohmann Δv = {dv_hoh:.3f} km/s (toi uu)")

for he_so_rb in [1.5, 2.0, 5.0]:
    rb = r_geo * he_so_rb
    _, _, _, dv_bie = bielliptic_delta_v(r_leo, r_geo, rb)
    thoi_gian_h = thoi_gian_bielliptic(r_leo, r_geo, rb)
    print(f"  Bi-elliptic (rb={he_so_rb}×r₂): Δv = {dv_bie:.3f} km/s, thoi gian = {thoi_gian_h:.1f} h")

print(f"\n  Ket luan: Hohmann thang cho LEO-GEO (R < 11.94)")
```

<section class="lesson-pitfalls">

## Cam bay

1. **R < 11.94 lang phi nhien lieu.** Bi-elliptic khong bao gio thang Hohmann duoi nguong nay; dung dung no.

2. **Chi phi thoi gian nghiem trong.** Thoi gian chuyen co the dai gap 3-10 lan; thuong khong dang tiet kiem delta-v.

3. **Ba lan dot = ba diem that bai.** Nhieu lan khoi dong lai dong co, nhieu rui ro.

4. **Phoi nhiem buc xa.** Vien diem cao di qua dai Van Allen; xau cho dien tu va phi hanh doan.

5. **r_b toi uu la vo han.** Trong thuc te, chon r_b dua tren rang buoc thoi gian, khong phai delta-v toi thieu.

6. **Bo qua nhieu loan.** Quy dao thuc can hieu chinh cho trong luc Mat Trang, Mat Troi.
