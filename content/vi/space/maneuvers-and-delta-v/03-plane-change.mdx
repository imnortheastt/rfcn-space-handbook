## Tai sao dieu nay quan trong

Thay doi mat phang la thao tac dat do nhat trong co hoc quy dao. Thay doi do nghieng chi vai do co the ton nhieu delta-v hon chuyen Hohmann den GEO. Day la ly do tai sao cac dia diem phong gan xich dao co gia tri—chung giam thieu thay doi do nghieng can thiet de dat quy dao xich dao. Hieu chi phi thay doi mat phang la can thiet cho lap ke hoach nhiem vu.

## Ap dung

**Tinh toan thay doi mat phang:**

```python
import numpy as np

MU_TRAI_DAT = 398600.4418  # km³/s²

def van_toc_tron(r, mu=MU_TRAI_DAT):
    """Van toc quy dao tron."""
    return np.sqrt(mu / r)

def thay_doi_mat_phang_dv(v, delta_i_deg):
    """
    Tinh delta-v cho thay doi mat phang don thuan.
    """
    delta_i_rad = np.radians(delta_i_deg)
    return 2 * v * np.sin(delta_i_rad / 2)

def thao_tac_ket_hop_dv(v1, v2, delta_i_deg):
    """
    Tinh delta-v cho thay doi do cao + mat phang ket hop.
    """
    delta_i_rad = np.radians(delta_i_deg)
    dv = np.sqrt(v1**2 + v2**2 - 2*v1*v2*np.cos(delta_i_rad))
    return dv

def do_cao_sang_ban_kinh(do_cao_km, r_trai_dat=6371):
    return do_cao_km + r_trai_dat

# Tinh toan vi du
print("Phan tich Thay doi Mat phang")
print("=" * 60)

# Chi phi thay doi mat phang tai cac do cao khac nhau
print("\n1. Chi phi Thay doi Mat phang theo Do cao")
print("-" * 60)

do_cao = [400, 2000, 10000, 20200, 35786]  # km
delta_i = 28.5  # do

print(f"  Thay doi do nghieng: {delta_i}°")
print(f"")
print(f"  {'Do cao':>12} {'Ban kinh':>10} {'Van toc':>10} {'Δv':>10}")
print(f"  {'(km)':>12} {'(km)':>10} {'(km/s)':>10} {'(km/s)':>10}")
print("-" * 60)

for alt in do_cao:
    r = do_cao_sang_ban_kinh(alt)
    v = van_toc_tron(r)
    dv = thay_doi_mat_phang_dv(v, delta_i)
    print(f"  {alt:>12} {r:>10.0f} {v:>10.2f} {dv:>10.2f}")

# Cac thay doi do nghieng khac nhau tai LEO
print("\n2. Chi phi Thay doi Mat phang tai LEO (400 km)")
print("-" * 60)

r_leo = do_cao_sang_ban_kinh(400)
v_leo = van_toc_tron(r_leo)

print(f"  Van toc LEO: {v_leo:.2f} km/s")
print(f"")
print(f"  {'Δi (°)':>10} {'Δv (km/s)':>12} {'Tuong duong':>25}")
print("-" * 60)

so_sanh = [
    (5, "Dieu chinh nho"),
    (10, "Thay doi dang ke"),
    (28.5, "Cape → Xich dao"),
    (45, "Trung binh → Cuc"),
    (90, "Xich dao → Cuc"),
    (180, "Lat nguoc"),
]

for delta_i, mo_ta in so_sanh:
    dv = thay_doi_mat_phang_dv(v_leo, delta_i)
    print(f"  {delta_i:>10.1f} {dv:>12.2f} {mo_ta:>25}")

# Cac chien luoc LEO den GEO
print("\n3. LEO (28.5°) den GEO (0°) — So sanh Chien luoc")
print("-" * 60)

r_leo = do_cao_sang_ban_kinh(400)
r_geo = do_cao_sang_ban_kinh(35786)
v_leo = van_toc_tron(r_leo)
v_geo = van_toc_tron(r_geo)
delta_i = 28.5

# Thong so GTO
a_gto = (r_leo + r_geo) / 2
v_gto_can_diem = np.sqrt(MU_TRAI_DAT * (2/r_leo - 1/a_gto))
v_gto_vien_diem = np.sqrt(MU_TRAI_DAT * (2/r_geo - 1/a_gto))

print(f"  LEO: r = {r_leo:.0f} km, v = {v_leo:.2f} km/s")
print(f"  GEO: r = {r_geo:.0f} km, v = {v_geo:.2f} km/s")
print(f"  GTO vien diem v = {v_gto_vien_diem:.2f} km/s")
print(f"")

# Chien luoc 1
dv1_mp = thay_doi_mat_phang_dv(v_leo, delta_i)
dv1_hoh1 = abs(v_gto_can_diem - v_leo)
dv1_hoh2 = abs(v_geo - v_gto_vien_diem)
dv1_tong = dv1_mp + dv1_hoh1 + dv1_hoh2

print("  Chien luoc 1: Thay doi mat phang tai LEO, roi Hohmann")
print(f"    Thay doi mat phang: {dv1_mp:.2f} km/s")
print(f"    Hohmann dot 1: {dv1_hoh1:.2f} km/s")
print(f"    Hohmann dot 2: {dv1_hoh2:.2f} km/s")
print(f"    TONG: {dv1_tong:.2f} km/s")
print()

# Chien luoc 2
dv2_hoh1 = abs(v_gto_can_diem - v_leo)
dv2_hoh2 = abs(v_geo - v_gto_vien_diem)
dv2_mp = thay_doi_mat_phang_dv(v_geo, delta_i)
dv2_tong = dv2_hoh1 + dv2_hoh2 + dv2_mp

print("  Chien luoc 2: Hohmann truoc, thay doi mat phang tai GEO")
print(f"    Hohmann dot 1: {dv2_hoh1:.2f} km/s")
print(f"    Hohmann dot 2: {dv2_hoh2:.2f} km/s")
print(f"    Thay doi mat phang: {dv2_mp:.2f} km/s")
print(f"    TONG: {dv2_tong:.2f} km/s")
print()

# Chien luoc 3
dv3_hoh1 = abs(v_gto_can_diem - v_leo)
dv3_ket_hop = thao_tac_ket_hop_dv(v_gto_vien_diem, v_geo, delta_i)
dv3_tong = dv3_hoh1 + dv3_ket_hop

print("  Chien luoc 3: Dot ket hop tai vien diem GTO (TOI UU)")
print(f"    Dot can diem: {dv3_hoh1:.2f} km/s")
print(f"    Dot ket hop vien diem: {dv3_ket_hop:.2f} km/s")
print(f"    TONG: {dv3_tong:.2f} km/s")
print()

tiet_kiem = (dv1_tong - dv3_tong) / dv1_tong * 100
print(f"  Tiet kiem so voi Chien luoc 1: {tiet_kiem:.1f}%")

# So sanh dia diem phong
print("\n4. So sanh Dia diem Phong cho Nhiem vu GEO")
print("-" * 60)

dia_diem_phong = [
    ("Kourou (ESA)", 5.2),
    ("Cape Canaveral", 28.5),
    ("Tanegashima", 30.4),
    ("Baikonur", 45.6),
    ("Plesetsk", 62.8),
]

print(f"  {'Dia diem':<20} {'Vi do':>10} {'Δv mat phang':>12} {'Tong GTO→GEO':>15}")
print("-" * 60)

for ten, vi_do in dia_diem_phong:
    dv_ket_hop = thao_tac_ket_hop_dv(v_gto_vien_diem, v_geo, vi_do)
    dv_mp_rieng = thay_doi_mat_phang_dv(v_geo, vi_do)
    print(f"  {ten:<20} {vi_do:>10.1f}° {dv_mp_rieng:>10.2f} km/s {dv_ket_hop:>13.2f} km/s")
```

## Cam bay

1. **Dot tai can diem de thay doi mat phang.** Dung bao gio thay doi mat phang tai can diem—van toc toi da, chi phi toi da.

2. **Dot rieng lang phi nhien lieu.** Luon ket hop thay doi do cao va mat phang khi co the.

3. **Danh gia thap chi phi.** Thay doi mat phang 45° tai LEO ton nhieu hon den GEO.

4. **Rang buoc dia diem phong.** Do nghieng toi thieu = vi do phong. Thao tac chan cho lang phi nhien lieu.

5. **Quen kinh do diem len.** Thay doi mat phang anh huong ca do nghieng VA RAAN; thoi diem quan trong.

6. **Tien dong dong bo mat troi.** Mot so do nghieng su dung nhieu loan J2 thay vi nhien lieu de quay RAAN.
