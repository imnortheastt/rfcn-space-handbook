## Ap dung

**Tinh toan hinh hoc vung an toan:**

```python
import numpy as np

def ellipsoid_boundary(semi_axes, num_points=100):
    """Tao cac diem tren be mat ellipsoid."""
    # Ban truc: (a, b, c) cho x, y, z
    a, b, c = semi_axes
    
    theta = np.linspace(0, 2*np.pi, num_points)
    phi = np.linspace(0, np.pi, num_points//2)
    
    theta, phi = np.meshgrid(theta, phi)
    
    x = a * np.sin(phi) * np.cos(theta)
    y = b * np.sin(phi) * np.sin(theta)
    z = c * np.cos(phi)
    
    return x, y, z

def point_in_ellipsoid(point, semi_axes):
    """Kiem tra diem co nam trong ellipsoid khong."""
    x, y, z = point
    a, b, c = semi_axes
    return (x/a)**2 + (y/b)**2 + (z/c)**2 <= 1

def point_in_corridor(point, half_angle_deg, corridor_axis='x'):
    """Kiem tra diem co nam trong hanh lang tiep can khong."""
    x, y, z = point
    half_angle = np.radians(half_angle_deg)
    
    if corridor_axis == 'x':  # Tiep can V-bar
        if x >= 0:  # Phai dang tiep can (x duong huong muc tieu)
            return False
        lateral_dist = np.sqrt(y**2 + z**2)
        axial_dist = abs(x)
        return lateral_dist <= axial_dist * np.tan(half_angle)
    return False

def check_zone(position_m, zones):
    """Xac dinh vi tri nam trong vung nao."""
    for zone_name, zone_def in sorted(zones.items(), 
                                       key=lambda x: x[1].get('priority', 0)):
        if zone_def['type'] == 'ellipsoid':
            if point_in_ellipsoid(position_m, zone_def['semi_axes']):
                return zone_name
        elif zone_def['type'] == 'sphere':
            if np.linalg.norm(position_m) <= zone_def['radius']:
                return zone_name
        elif zone_def['type'] == 'corridor':
            if point_in_corridor(position_m, zone_def['half_angle']):
                return zone_name
    return 'ngoai'

def gate_criteria(from_zone, to_zone, state):
    """Kiem tra tieu chi cong cho chuyen tiep vung."""
    range_m = state['range_m']
    range_rate = state['range_rate_m_s']
    lateral_rate = state['lateral_rate_m_s']
    
    criteria = {
        ('zone4', 'zone3'): {
            'range_rate_max': 10.0,
            'lateral_rate_max': 2.0,
            'nav_converged': True,
            'comm_verified': True,
        },
        ('zone3', 'zone2'): {
            'range_rate_max': 2.0,
            'lateral_rate_max': 0.5,
            'corridor_offset_max': 50.0,
            'cam_fuel_min': 10.0,
        },
        ('zone2', 'zone1'): {
            'range_rate_max': 0.5,
            'lateral_rate_max': 0.1,
            'alignment_max_deg': 3.0,
            'lidar_locked': True,
        },
    }
    
    key = (from_zone, to_zone)
    if key not in criteria:
        return True, []
    
    c = criteria[key]
    failures = []
    
    if range_rate > c.get('range_rate_max', float('inf')):
        failures.append(f"toc_do_tam {range_rate:.1f} > {c['range_rate_max']}")
    if lateral_rate > c.get('lateral_rate_max', float('inf')):
        failures.append(f"toc_do_ngang {lateral_rate:.2f} > {c['lateral_rate_max']}")
    
    return len(failures) == 0, failures

# Dinh nghia vung ISS
ISS_ZONES = {
    'zone1': {'type': 'corridor', 'half_angle': 10, 'priority': 1},
    'zone2': {'type': 'sphere', 'radius': 200, 'priority': 2},
    'zone3': {'type': 'ellipsoid', 'semi_axes': (2000, 1500, 1500), 'priority': 3},
    'zone4': {'type': 'ellipsoid', 'semi_axes': (4000, 3000, 3000), 'priority': 4},
}

# Phan tich
print("Phan tich Vung An toan")
print("=" * 60)

# Vi tri kiem tra
test_positions = [
    ([-5000, 0, 0], "5 km tren V-bar"),
    ([-3000, 500, 0], "3 km, lech ngang 500m"),
    ([-1500, 0, 0], "1.5 km tren V-bar"),
    ([-150, 0, 0], "150 m tren V-bar"),
    ([-100, 50, 0], "100 m, lech 50m"),
]

print("\nPhan loai Vung Vi tri:")
print("-" * 60)
for pos, desc in test_positions:
    zone = check_zone(pos, ISS_ZONES)
    print(f"  {desc:30} → {zone}")

# Kiem tra chuyen tiep cong
print("\n" + "-" * 60)
print("Phan tich Chuyen tiep Cong:")

states = [
    {'range_m': 4000, 'range_rate_m_s': 8.0, 'lateral_rate_m_s': 1.0},
    {'range_m': 4000, 'range_rate_m_s': 15.0, 'lateral_rate_m_s': 1.0},
    {'range_m': 2000, 'range_rate_m_s': 1.5, 'lateral_rate_m_s': 0.3},
    {'range_m': 200, 'range_rate_m_s': 0.3, 'lateral_rate_m_s': 0.05},
]

transitions = [
    ('zone4', 'zone3'),
    ('zone4', 'zone3'),
    ('zone3', 'zone2'),
    ('zone2', 'zone1'),
]

print(f"\n  {'Chuyen tiep':20} {'Tam':>8} {'Toc do':>10} {'Ket qua':>10}")
print(f"  {'-'*20} {'-'*8} {'-'*10} {'-'*10}")
for (from_z, to_z), state in zip(transitions, states):
    passed, failures = gate_criteria(from_z, to_z, state)
    result = "DAT" if passed else "KHONG"
    print(f"  {from_z}→{to_z:12} {state['range_m']:>7}m "
          f"{state['range_rate_m_s']:>8.1f}m/s {result:>10}")
    if failures:
        for f in failures:
            print(f"    ⚠ {f}")

# Loai tru luong phun
print("\n" + "-" * 60)
print("Vung Loai tru Luong phun:")
exclusion_cones = [
    {'name': 'Tam Pin MT', 'half_angle': 30, 'min_range': 50},
    {'name': 'Tam Tan Nhiet', 'half_angle': 20, 'min_range': 30},
    {'name': 'Anten', 'half_angle': 15, 'min_range': 100},
]

for cone in exclusion_cones:
    solid_angle = 2 * np.pi * (1 - np.cos(np.radians(cone['half_angle'])))
    fraction = solid_angle / (4 * np.pi) * 100
    print(f"  {cone['name']:15} ±{cone['half_angle']}° "
          f"({fraction:.1f}% mat cau), toi thieu {cone['min_range']}m")
```

<section class="lesson-pitfalls">

## Cam bay

1. **Nham lan ellipsoid vs mat cau.** Vung truong xa thuong la ellipsoid thang hang voi vector van toc, khong phai mat cau.

2. **Nua goc hanh lang thay doi.** Cac cong cap tau khac nhau co the co hinh hoc hanh lang khac nhau.

3. **Tieu chi cong la dieu kien VA.** Tat ca tieu chi phai dat; mot that bai chan chuyen tiep.

4. **Ranh gioi vung la dong.** Trong khi ISS day len hoac thay doi tu the, vung co the tam thoi mo rong.

5. **Vung luong phun phu thuoc lua chon day.** Day nao dot xac dinh loai tru nao ap dung.

6. **Vung RF la hai chieu.** Ca bo phat tau tham va tram tao vung loai tru.

7. **Anh sang anh huong van hanh vung.** Mot so cam bien quang hoc can anh sang ban ngay; chuyen tiep vung co the bi rang buoc anh sang.
