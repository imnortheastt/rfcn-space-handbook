**Thoi gian va delta-V giai doan RPOD:**

```python
import numpy as np

# Hang so
MU_EARTH = 398600.4418  # kmВі/sВІ

def compute_orbital_params(altitude_km):
    """Tinh tham so quy dao co ban."""
    r = 6371 + altitude_km
    v = np.sqrt(MU_EARTH / r)
    T = 2 * np.pi * np.sqrt(r**3 / MU_EARTH)
    n = 2 * np.pi / T
    return {
        'radius_km': r,
        'velocity_km_s': v,
        'period_s': T,
        'period_min': T / 60,
        'mean_motion_rad_s': n
    }

def phasing_maneuver(phase_angle_deg, target_altitude_km, num_orbits=1):
    """Tinh delta-V cho dinh pha de giam goc pha."""
    # Goc pha: bao xa phia sau (duong) hoac truoc (am)
    r_target = 6371 + target_altitude_km
    T_target = 2 * np.pi * np.sqrt(r_target**3 / MU_EARTH)
    
    # Chu ky quy dao dinh pha
    phase_angle_rad = np.radians(phase_angle_deg)
    phase_to_cover = 2 * np.pi * num_orbits - phase_angle_rad
    T_phasing = T_target * phase_to_cover / (2 * np.pi * num_orbits)
    
    # Ban truc lon quy dao dinh pha
    a_phasing = (MU_EARTH * (T_phasing / (2 * np.pi))**2) ** (1/3)
    
    # Delta-V (kieu Hohmann)
    v_target = np.sqrt(MU_EARTH / r_target)
    v_phasing_apo = np.sqrt(MU_EARTH * (2/r_target - 1/a_phasing))
    
    dv_enter = abs(v_phasing_apo - v_target)
    dv_exit = dv_enter  # Doi xung khi quay lai
    
    return {
        'phase_angle_deg': phase_angle_deg,
        'num_orbits': num_orbits,
        'phasing_period_min': T_phasing / 60,
        'target_period_min': T_target / 60,
        'delta_v_total_m_s': (dv_enter + dv_exit) * 1000,
        'phasing_altitude_km': (2 * a_phasing - r_target) - 6371
    }

def approach_delta_v(ranges_km, closure_rates_m_s):
    """Uoc tinh delta-V cho dong tac tiep can."""
    # Don gian hoa: gia su moi thay doi tam can tang toc + giam toc
    total_dv = 0
    maneuvers = []
    
    for i in range(len(ranges_km) - 1):
        # Tang toc den toc do dong, roi giam toc den toc do cham hon
        if i < len(closure_rates_m_s) - 1:
            dv = abs(closure_rates_m_s[i] - closure_rates_m_s[i+1])
        else:
            dv = closure_rates_m_s[i]
        
        total_dv += dv * 2  # Tang toc + giam toc
        maneuvers.append({
            'from_km': ranges_km[i],
            'to_km': ranges_km[i+1],
            'closure_m_s': closure_rates_m_s[i],
            'dv_m_s': dv * 2
        })
    
    return total_dv, maneuvers

def capture_requirements():
    """Yeu cau giao dien bat giu dien hinh."""
    return {
        'max_closing_velocity_m_s': 0.10,  # 10 cm/s
        'max_lateral_velocity_m_s': 0.05,  # 5 cm/s
        'max_angular_rate_deg_s': 0.5,
        'max_lateral_misalignment_m': 0.15,  # 15 cm
        'max_angular_misalignment_deg': 5.0,
        'capture_envelope_diameter_m': 0.30,  # 30 cm
    }

# Phan tich
print("Phan tich Hoat dong RPOD")
print("=" * 60)

altitude = 420  # km (ISS)

print(f"\nQuy dao Muc tieu (ISS):")
print(f"  Do cao: {altitude} km")
print(f"  Van toc: {orbit['velocity_km_s']:.3f} km/s")
print(f"  Chu ky: {orbit['period_min']:.1f} phut")

# Dinh pha
print("\n" + "-" * 60)
print("Giai doan 1: Dong tac Dinh pha")
phase = phasing_maneuver(90, altitude, num_orbits=2)  # 90В° phia sau
print(f"  Goc pha ban dau: {phase['phase_angle_deg']}В° phia sau")
print(f"  Bat kip trong: {phase['num_orbits']} vong quy dao")
print(f"  Chu ky quy dao dinh pha: {phase['phasing_period_min']:.1f} phut")
print(f"  Do cao quy dao dinh pha: {phase['phasing_altitude_km']:.0f} km")
print(f"  Tong delta-V: {phase['delta_v_total_m_s']:.1f} m/s")

# Giai doan tiep can
print("\n" + "-" * 60)
print("Giai doan 2-4: Tiep can")
ranges = [50, 5, 1.5, 0.4, 0.2, 0.03, 0.01, 0]  # km
closure_rates = [10, 5, 2, 0.5, 0.2, 0.1, 0.03]  # m/s
total_dv, maneuvers = approach_delta_v(ranges, closure_rates)

print(f"\n  Diem tham   Tam      Dong      Uoc dV")
print(f"  в”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җ")
for i, m in enumerate(maneuvers):
    print(f"  WP{i+1}        {m['from_km']*1000:6.0f}m  {m['closure_m_s']:5.2f}m/s  {m['dv_m_s']:5.2f}m/s")

print(f"\n  Tong delta-V tiep can: ~{total_dv:.1f} m/s")

# Yeu cau bat giu
print("\n" + "-" * 60)
print("Giai doan 5: Yeu cau Bat giu")
capture = capture_requirements()
print(f"  Van toc dong toi da: {capture['max_closing_velocity_m_s']*100:.0f} cm/s")
print(f"  Van toc ngang toi da: {capture['max_lateral_velocity_m_s']*100:.0f} cm/s")
print(f"  Toc do goc toi da: {capture['max_angular_rate_deg_s']}В°/s")
print(f"  Do lech ngang toi da: {capture['max_lateral_misalignment_m']*100:.0f} cm")
print(f"  Do lech goc toi da: {capture['max_angular_misalignment_deg']}В°")

# Tong delta-V nhiem vu
print("\n" + "-" * 60)
print("Tong Ngan sach Delta-V RPOD:")
total_rpod = phase['delta_v_total_m_s'] + total_dv
print(f"  Dinh pha: {phase['delta_v_total_m_s']:.1f} m/s")
```

## Cam bay

1. **Khoang trong chuyen giao cam bien.** Chuyen doi giua cam bien co the gay gian doan dan duong; can vung phu chong lan.

2. **Rang buoc anh sang.** Cam bien quang hoc khong hoat dong trong bong toi; lap ke hoach tiep can trong ngay quy dao.

3. **Mat lien lac mat dat.** Mat lien lac mat dat trong giai doan quan trong can tu dong tren tau.

4. **Du tru nhien lieu huy bo.** CAM phai luon san sang; khong tieu thu du phong khi tiep can.

5. **Khong chac chan tu the muc tieu.** Tu the muc tieu quay hoac bat thuong lam phuc tap bat giu.

6. **Va dap luong phun.** Dot phun RCS co the lam hong tam pin mat troi hoac cam bien muc tieu.

7. **Da duong GPS tuong doi.** Gan cau truc lon, do chinh xac GPS giam do phan xa.
