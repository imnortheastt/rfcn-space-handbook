## Tai sao dieu nay quan trong

Khi tau vu tru tiep can ISS hoac xe phuc vu tiep can ve tinh, phi cong khong nghi theo toa do tuyet doi—ho nghi theo chuyen dong tuong doi. Phuong trinh Clohessy-Wiltshire (CW) mo ta cach "tau duoi" di chuyen tuong doi voi "muc tieu" trong quy dao tron gan ke. Mo hinh tuyen tinh hoa nay la nen tang cua moi hoat dong can ke, tu cap tau Gemini den phuc vu ve tinh hien dai.

## Ap dung

**Trien khai phuong trinh Clohessy-Wiltshire:**

```python
import numpy as np
import matplotlib.pyplot as plt

# Hang so
MU_EARTH = 398600.4418  # km³/s²

def cw_state_transition(t, n):
    """Tinh ma tran chuyen doi trang thai CW Φ(t).
    
    Vecto trang thai: [x, y, z, ẋ, ẏ, ż]
    """
    s = np.sin(n * t)
    c = np.cos(n * t)
    
    # Ma tran chuyen doi trang thai 6x6
    phi = np.array([
        [4 - 3*c,      0, 0,     s/n,        2*(1-c)/n,    0],
        [6*(s - n*t),  1, 0,    -2*(1-c)/n,  (4*s - 3*n*t)/n, 0],
        [0,            0, c,     0,          0,            s/n],
        [3*n*s,        0, 0,     c,          2*s,          0],
        [-6*n*(1-c),   0, 0,    -2*s,        4*c - 3,      0],
        [0,            0, -n*s,  0,          0,            c]
    ])
    
    return phi

def propagate_cw(state0, t, n):
    """Lan truyen trang thai dung phuong trinh CW."""
    phi = cw_state_transition(t, n)
    return phi @ state0

def compute_mean_motion(altitude_km):
    """Tinh chuyen dong trung binh quy dao."""
    r = 6371 + altitude_km  # km
    return np.sqrt(MU_EARTH / r**3)  # rad/s

def simulate_relative_motion(state0, n, duration_orbits=2, dt=60):
    """Mo phong quy dao chuyen dong tuong doi."""
    T = 2 * np.pi / n  # chu ky quy dao
    t_span = np.arange(0, duration_orbits * T, dt)
    
    trajectory = np.array([propagate_cw(state0, t, n) for t in t_span])
    
    return t_span, trajectory

def delta_v_for_rendezvous(r0, v0, r_target, t_transfer, n):
    """Tinh delta-V cho gap go hai xung.
    
    Tu vi tri hien tai r0 voi van toc v0,
    den r_target trong thoi gian t_transfer.
    """
    phi = cw_state_transition(t_transfer, n)
    
    # Phan chia ma tran chuyen doi trang thai
    phi_rr = phi[:3, :3]
    phi_rv = phi[:3, 3:]
    phi_vr = phi[3:, :3]
    phi_vv = phi[3:, 3:]
    
    # Van toc can thiet tai diem bat dau
    v_required = np.linalg.inv(phi_rv) @ (r_target - phi_rr @ r0)
    
    # Delta-V tai diem khoi hanh
    dv1 = v_required - v0
    
    # Van toc tai diem den
    v_arrival = phi_vr @ r0 + phi_vv @ v_required
    
    # Delta-V tai diem den (de dung chuyen dong tuong doi)
    dv2 = -v_arrival
    
    return dv1, dv2, np.linalg.norm(dv1) + np.linalg.norm(dv2)

# Vi du: Hoat dong can ke ISS
print("Phan tich Clohessy-Wiltshire")
print("=" * 60)

# Quy dao ISS
altitude = 420  # km
n = compute_mean_motion(altitude)
T = 2 * np.pi / n
print(f"\nQuy dao ISS:")
print(f"  Do cao: {altitude} km")
print(f"  Chuyen dong trung binh: {n*1000:.4f} mrad/s")
print(f"  Chu ky quy dao: {T/60:.1f} phut")

# Kich ban 1: Quy dao bong bau duc
print("\n" + "-" * 60)
print("Kich ban 1: Quy dao 'Bong bau duc' Khep kin")
x0, y0, z0 = 100, 0, 0  # Bat dau 100m phia truoc
# Cho chuyen dong bi chan: ẏ₀ = -2n x₀
vx0, vy0, vz0 = 0, -2*n*x0, 0  # m/s da chuyen doi
state_football = np.array([x0, y0, z0, vx0, vy0, vz0]) / 1000  # km

t, traj = simulate_relative_motion(state_football, n, duration_orbits=1.5)
print(f"  Ban dau: x={x0}m, ẏ={vy0*1000:.3f} m/s")
print(f"  Chuyen dong: elip bi chan, ti le 2:1")
print(f"  Do rong y toi da: {np.max(np.abs(traj[:,1]))*1000:.1f} m")

# Kich ban 2: Tiep can V-bar
print("\n" + "-" * 60)
print("Kich ban 2: Tiep can V-bar (tu phia sau)")
# Bat dau tren V-bar (x=0), 500m phia sau
state_vbar = np.array([0, -0.5, 0, 0, 0, 0])  # km, km/s

# Tinh delta-V de den goc toa do trong nua vong quy dao
t_transfer = T / 2
dv1, dv2, total_dv = delta_v_for_rendezvous(
    state_vbar[:3], state_vbar[3:], 
    np.array([0, 0, 0]), 
    t_transfer, n
)
print(f"  Bat dau: 500m phia sau tren V-bar")
print(f"  Thoi gian chuyen tiep: {t_transfer/60:.1f} phut (nua vong)")
print(f"  Delta-V 1: {np.linalg.norm(dv1)*1000:.2f} m/s")
print(f"  Delta-V 2: {np.linalg.norm(dv2)*1000:.2f} m/s")
print(f"  Tong delta-V: {total_dv*1000:.2f} m/s")

# Kich ban 3: Nhay R-bar
print("\n" + "-" * 60)
print("Kich ban 3: Nhay R-bar (dong tac huong kinh)")
# Bat dau tai goc toa do, ap dung delta-V huong kinh nho
```

## Cam bay

1. **Gia thiet quy dao tron.** Phuong trinh CW khong dung cho quy dao muc tieu lech tam; dung phuong trinh Tschauner-Hempel thay the.
