## Tai sao dieu nay quan trong

Khi ve tinh khong the nhin thay tram mat dat, no chuyen tiep du lieu bang cach nao? Lien ket lien ve tinh (ISL) cho phep tau vu tru giao tiep truc tiep voi nhau, tao mang luoi tren quy dao. ISL dua tren RF la cong nghe chin muoi, chiu duoc thoi tiet tao xuong song cua cac chom sao chuyen tiep nhu TDRSS va cho phep ket noi cheo trong cac he thong LEO thuong mai. Hieu thiet ke RF ISL la can thiet cho kien truc su chom sao va nguoi lap ke hoach phan mat dat.

## Ap dung

**May tinh ngan sach lien ket RF ISL:**

```python
import numpy as np

# Hang so
C = 299792458  # m/s
K_BOLTZMANN = 1.380649e-23  # J/K

def fspl_db(distance_km, frequency_ghz):
    """Ton hao khong gian tu do tinh bang dB."""
    d_m = distance_km * 1000
    f_hz = frequency_ghz * 1e9
    return 20 * np.log10(4 * np.pi * d_m * f_hz / C)

def antenna_gain_db(diameter_m, frequency_ghz, efficiency=0.55):
    """Tang ich anten parabol."""
    wavelength = C / (frequency_ghz * 1e9)
    area = np.pi * (diameter_m / 2) ** 2
    gain = efficiency * 4 * np.pi * area / wavelength**2
    return 10 * np.log10(gain)

def beamwidth_deg(diameter_m, frequency_ghz):
    """Do rong buc xa nua cong suat (do)."""
    wavelength = C / (frequency_ghz * 1e9)
    return np.degrees(70 * wavelength / diameter_m)

def noise_power_dbm(bandwidth_hz, noise_temp_k):
    """Cong suat nhieu tinh bang dBm."""
    noise_w = K_BOLTZMANN * noise_temp_k * bandwidth_hz
    return 10 * np.log10(noise_w) + 30  # Chuyen doi sang dBm

def rf_isl_link_budget(tx_power_dbm, tx_ant_m, rx_ant_m, 
                       freq_ghz, range_km, bandwidth_mhz,
                       tx_loss_db=1, rx_loss_db=1, rx_temp_k=500):
    """Ngan sach lien ket RF ISL day du."""
    # Tang ich
    tx_gain = antenna_gain_db(tx_ant_m, freq_ghz)
    rx_gain = antenna_gain_db(rx_ant_m, freq_ghz)
    
    # EIRP
    eirp = tx_power_dbm + tx_gain - tx_loss_db
    
    # Ton hao duong truyen
    fspl = fspl_db(range_km, freq_ghz)
    
    # Cong suat nhan
    rx_power = eirp - fspl + rx_gain - rx_loss_db
    
    # Nhieu
    bw_hz = bandwidth_mhz * 1e6
    noise = noise_power_dbm(bw_hz, rx_temp_k)
    
    # C/N
    cn_db = rx_power - noise
    
    # G/T
    gt = rx_gain - 10 * np.log10(rx_temp_k)
    
    # Do rong buc xa
    bw_tx = beamwidth_deg(tx_ant_m, freq_ghz)
    bw_rx = beamwidth_deg(rx_ant_m, freq_ghz)
    
    return {
        'tx_power_dbm': tx_power_dbm,
        'tx_gain_dbi': tx_gain,
        'rx_gain_dbi': rx_gain,
        'eirp_dbm': eirp,
        'fspl_db': fspl,
        'rx_power_dbm': rx_power,
        'noise_floor_dbm': noise,
        'cn_db': cn_db,
        'gt_dbk': gt,
        'tx_beamwidth_deg': bw_tx,
        'rx_beamwidth_deg': bw_rx,
        'pointing_req_deg': bw_tx / 10,
    }

def estimate_data_rate(cn_db, bandwidth_mhz, modulation='QPSK'):
    """Uoc tinh toc do du lieu dat duoc."""
    spectral_eff = {
        'BPSK': 1,
        'QPSK': 2,
        '8PSK': 3,
        '16QAM': 4,
        '64QAM': 6,
    }
    cn_req = {
        'BPSK': 6,
        'QPSK': 10,
        '8PSK': 14,
        '16QAM': 17,
        '64QAM': 23,
    }
    
    if cn_db >= cn_req[modulation]:
        rate_mbps = bandwidth_mhz * spectral_eff[modulation] * 0.8  # Hieu suat 80%
        margin = cn_db - cn_req[modulation]
        return rate_mbps, margin
    return 0, cn_db - cn_req[modulation]

# Phan tich
print("Phan tich Ngan sach Lien ket RF ISL")
print("=" * 60)

# Kich ban 1: LEO ISL trong mat phang (Ka-band)
print("\n1. LEO ISL Trong Mat phang (Ka-band, 2000 km)")
print("-" * 60)
budget1 = rf_isl_link_budget(
    tx_power_dbm=37,      # 5 W
    tx_ant_m=0.5,
    rx_ant_m=0.5,
    freq_ghz=25,
    range_km=2000,
    bandwidth_mhz=100,
)

for k, v in budget1.items():
    if 'db' in k.lower() or 'dbi' in k.lower() or 'dbk' in k.lower() or 'dbm' in k.lower():
        print(f"  {k:20}: {v:+.1f} dB")
    elif 'deg' in k.lower():
        print(f"  {k:20}: {v:.2f}°")
    else:
        print(f"  {k:20}: {v:.1f}")

rate, margin = estimate_data_rate(budget1['cn_db'], 100, '16QAM')
print(f"\n  Toc do dat duoc (16QAM): {rate:.0f} Mbps, bien do: {margin:.1f} dB")

# Kich ban 2: LEO ISL giua mat phang (tam xa hon)
print("\n2. LEO ISL Giua Mat phang (V-band, 5000 km)")
print("-" * 60)
budget2 = rf_isl_link_budget(
    tx_power_dbm=40,      # 10 W
    tx_ant_m=0.3,         # Anten nho hon cho V-band
    rx_ant_m=0.3,
    freq_ghz=60,          # V-band
    range_km=5000,
    bandwidth_mhz=500,
)

for k, v in budget2.items():
    if 'db' in k.lower() or 'dbi' in k.lower() or 'dbk' in k.lower() or 'dbm' in k.lower():
        print(f"  {k:20}: {v:+.1f} dB")
    elif 'deg' in k.lower():
        print(f"  {k:20}: {v:.2f}°")
    else:
        print(f"  {k:20}: {v:.1f}")

rate2, margin2 = estimate_data_rate(budget2['cn_db'], 500, 'QPSK')
print(f"\n  Toc do dat duoc (QPSK): {rate2:.0f} Mbps, bien do: {margin2:.1f} dB")

# Bang so sanh
print("\n" + "=" * 60)
print("So sanh Bang RF ISL (lien ket 2000 km)")
print("-" * 60)
bands = [
    ('Ku', 15, 0.6),
    ('Ka', 25, 0.5),
    ('V', 60, 0.3),
]

print(f"{'Bang':6} {'Tan so':>8} {'Dia':>8} {'FSPL':>10} {'Tang ich':>8} {'BW':>8}")
print(f"{'':6} {'(GHz)':>8} {'(m)':>8} {'(dB)':>10} {'(dBi)':>8} {'(do)':>8}")
print("-" * 60)
for name, freq, dish in bands:
    fspl = fspl_db(2000, freq)
    gain = antenna_gain_db(dish, freq)
    bw = beamwidth_deg(dish, freq)
    print(f"{name:6} {freq:>8.0f} {dish:>8.1f} {fspl:>10.1f} {gain:>8.1f} {bw:>8.2f}")
```

<section class="lesson-pitfalls">

## Cam bay

1. **Loi chi huong anten.** Chi 0.1° loi chi huong tai Ka-band co the mat 3 dB — quan trong cho bien do.

2. **Van de duong noi giua mat phang.** Tai vung cuc, mat phang ke phan ky; ISL giua mat phang co the can chuyen giao.

3. **Bien doi tam.** Tam ISL giua mat phang co the thay doi 2:1 trong mot quy dao — ngan sach lien ket phai xu ly truong hop xau nhat.

4. **Dich chuyen Doppler.** LEO ISL thay ±50 kHz Doppler tai Ku-band; may thu can vong khoa theo doi.

5. **Phoi hop quy dinh.** Tan so ISL van can phoi hop ITU ngay ca trong khong gian.

6. **Chu ky nhiet nguyet thuc.** Dien tu RF phai song sot dao dong -40°C den +80°C moi vong quy dao.

7. **Nhieu tu Mat Troi.** Khi Mat Troi o trong boresight anten, nhiet do nhieu tang dot bien.
