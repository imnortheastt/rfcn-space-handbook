## Why this matters

Every spacecraft position and velocity must be expressed in some coordinate system. Choose wrong, and your orbit propagation will silently drift, your collision avoidance will fail, and your antenna pointing will miss the target. ECI (Earth-Centered Inertial) is the foundational frame for orbital mechanics — it doesn't rotate with Earth, so Newton's laws apply directly without fictitious forces. Understanding ECI is prerequisite to reading TLEs, computing orbital elements, and transforming between the many frames used in space operations.

## Apply

**Working with ECI coordinates:**

```python
import numpy as np
from datetime import datetime

# Earth parameters
MU_EARTH = 398600.4418  # km³/s² (gravitational parameter)
R_EARTH = 6378.137      # km (equatorial radius)
OMEGA_EARTH = 7.292115e-5  # rad/s (rotation rate)

def state_to_orbital_elements(r, v, mu=MU_EARTH):
    """Convert ECI state vector to classical orbital elements."""
    r_mag = np.linalg.norm(r)
    v_mag = np.linalg.norm(v)
    
    # Specific angular momentum
    h = np.cross(r, v)
    h_mag = np.linalg.norm(h)
    
    # Node vector (points to ascending node)
    k = np.array([0, 0, 1])
    n = np.cross(k, h)
    n_mag = np.linalg.norm(n)
    
    # Eccentricity vector
    e_vec = ((v_mag**2 - mu/r_mag) * r - np.dot(r, v) * v) / mu
    e = np.linalg.norm(e_vec)
    
    # Semi-major axis
    energy = v_mag**2 / 2 - mu / r_mag
    if abs(e - 1.0) > 1e-10:
        a = -mu / (2 * energy)
    else:
        a = float('inf')  # Parabolic
    
    # Inclination
    i = np.arccos(h[2] / h_mag)
    
    # RAAN (Right Ascension of Ascending Node)
    if n_mag > 1e-10:
        raan = np.arccos(n[0] / n_mag)
        if n[1] < 0:
            raan = 2 * np.pi - raan
    else:
        raan = 0.0
    
    # Argument of perigee
    if n_mag > 1e-10 and e > 1e-10:
        omega = np.arccos(np.dot(n, e_vec) / (n_mag * e))
        if e_vec[2] < 0:
            omega = 2 * np.pi - omega
    else:
        omega = 0.0
    
    # True anomaly
    if e > 1e-10:
        nu = np.arccos(np.dot(e_vec, r) / (e * r_mag))
        if np.dot(r, v) < 0:
            nu = 2 * np.pi - nu
    else:
        nu = 0.0
    
    return {
        'a': a,          # Semi-major axis (km)
        'e': e,          # Eccentricity
        'i': np.degrees(i),     # Inclination (deg)
        'raan': np.degrees(raan),  # RAAN (deg)
        'omega': np.degrees(omega), # Argument of perigee (deg)
        'nu': np.degrees(nu)    # True anomaly (deg)
    }

def orbital_period(a, mu=MU_EARTH):
    """Calculate orbital period in minutes."""
    return 2 * np.pi * np.sqrt(a**3 / mu) / 60

def greenwich_sidereal_time(jd):
    """Calculate Greenwich Mean Sidereal Time (radians)."""
    T = (jd - 2451545.0) / 36525  # Julian centuries from J2000
    gmst_deg = 280.46061837 + 360.98564736629 * (jd - 2451545.0) + \
               0.000387933 * T**2 - T**3 / 38710000
    return np.radians(gmst_deg % 360)

def eci_to_ecef(r_eci, jd):
    """Convert ECI position to ECEF."""
    gst = greenwich_sidereal_time(jd)
    cos_gst = np.cos(gst)
    sin_gst = np.sin(gst)
    
    R = np.array([
        [cos_gst, sin_gst, 0],
        [-sin_gst, cos_gst, 0],
        [0, 0, 1]
    ])
    
    return R @ r_eci

print("ECI Reference Frame Examples")
print("=" * 70)

# Example: ISS-like orbit
print("\n1. ISS State Vector (J2000 ECI)")
print("-" * 70)

r_eci = np.array([-2436.45, -2436.45, 6891.04])  # km
v_eci = np.array([-5.654, 4.830, 2.312])         # km/s

print(f"  Position: [{r_eci[0]:.2f}, {r_eci[1]:.2f}, {r_eci[2]:.2f}] km")
print(f"  Velocity: [{v_eci[0]:.3f}, {v_eci[1]:.3f}, {v_eci[2]:.3f}] km/s")

r_mag = np.linalg.norm(r_eci)
v_mag = np.linalg.norm(v_eci)
altitude = r_mag - R_EARTH

print(f"\n  Radius: {r_mag:.2f} km")
print(f"  Altitude: {altitude:.2f} km")
print(f"  Speed: {v_mag:.3f} km/s")

# Convert to orbital elements
print("\n2. Classical Orbital Elements")
print("-" * 70)

elements = state_to_orbital_elements(r_eci, v_eci)
period = orbital_period(elements['a'])

print(f"  Semi-major axis (a): {elements['a']:.2f} km")
print(f"  Eccentricity (e): {elements['e']:.6f}")
print(f"  Inclination (i): {elements['i']:.2f}°")
print(f"  RAAN (Ω): {elements['raan']:.2f}°")
print(f"  Arg of perigee (ω): {elements['omega']:.2f}°")
print(f"  True anomaly (ν): {elements['nu']:.2f}°")
print(f"  Orbital period: {period:.2f} minutes")

# ECI to ECEF conversion
print("\n3. ECI to ECEF Conversion")
print("-" * 70)

jd = 2451545.0 + 26 * 365.25 + 154  # Approximate JD for 2026-06-03
r_ecef = eci_to_ecef(r_eci, jd)

print(f"  Julian Date: {jd:.1f}")
print(f"  GST: {np.degrees(greenwich_sidereal_time(jd)):.2f}°")
print(f"\n  ECI:  [{r_eci[0]:.2f}, {r_eci[1]:.2f}, {r_eci[2]:.2f}] km")
print(f"  ECEF: [{r_ecef[0]:.2f}, {r_ecef[1]:.2f}, {r_ecef[2]:.2f}] km")

# Frame comparison
print("\n4. Why Frame Choice Matters")
print("-" * 70)

print("  Frame offset examples (typical values):")
print()
print(f"  {'Comparison':<25} {'Angular Offset':<18} {'Position Error (LEO)'}")
print("-" * 70)
print(f"  {'J2000 vs GCRF':<25} {'~0.02 arcsec':<18} {'~0.1 m'}")
print(f"  {'J2000 vs TEME':<25} {'~20 arcsec':<18} {'~600 m'}")
print(f"  {'J2000 vs B1950':<25} {'~0.5°':<18} {'~50 km'}")

print("\n  CRITICAL: Match frame to propagator!")
print("    - TLE/SGP4 → TEME frame")
print("    - Precision ephemeris → J2000/GCRF")
print("    - GPS → WGS84 (which is ECEF)")

# Precession visualization
print("\n5. Earth's Axial Precession")
print("-" * 70)

prec_rate = 50.3  # arcsec/year
years = [0, 10, 50, 100, 1000]

print("  Years from J2000.0 │ Precession angle │ Position shift (GEO)")
print("-" * 70)
for y in years:
    angle = prec_rate * y / 3600  # degrees
    shift = 42164 * np.sin(np.radians(angle))  # km at GEO
    print(f"  {y:>19} │ {angle:>14.4f}° │ {shift:>14.1f} km")

print("\n  Without epoch specification, positions become ambiguous over time!")
```

<section class="lesson-pitfalls">

## Pitfalls

1. **Mixing ECI variants.** J2000, TEME, TOD are all "ECI" but differ by arcseconds to arcminutes. Always specify which.

2. **Forgetting epoch.** ECI without epoch is ambiguous. Always pair coordinates with time.

3. **Using ECI for ground stations.** Ground antennas don't move in ECEF, not ECI. Transform before computing pointing.

4. **TLE frame mismatch.** TLEs are in TEME. Converting SGP4 output to J2000 without proper transformation loses accuracy.

5. **Ignoring polar motion.** For sub-meter accuracy, account for Earth orientation parameters (EOP).

6. **Assuming perfect inertial.** ECI precesses slowly. For multi-decade analysis, use GCRF or apply precession models.
