## Why this matters

When a satellite can't see a ground station, how does it relay data? Inter-satellite links (ISLs) let spacecraft communicate directly with each other, creating mesh networks in orbit. RF-based ISLs are mature, weather-tolerant technology that forms the backbone of relay constellations like TDRSS and enables crosslinks in commercial LEO systems. Understanding RF ISL design is essential for constellation architects and ground segment planners.

## Apply

**RF ISL link budget calculator:**

```python
import numpy as np

# Constants
C = 299792458  # m/s
K_BOLTZMANN = 1.380649e-23  # J/K

def fspl_db(distance_km, frequency_ghz):
    """Free space path loss in dB."""
    d_m = distance_km * 1000
    f_hz = frequency_ghz * 1e9
    return 20 * np.log10(4 * np.pi * d_m * f_hz / C)

def antenna_gain_db(diameter_m, frequency_ghz, efficiency=0.55):
    """Parabolic antenna gain."""
    wavelength = C / (frequency_ghz * 1e9)
    area = np.pi * (diameter_m / 2) ** 2
    gain = efficiency * 4 * np.pi * area / wavelength**2
    return 10 * np.log10(gain)

def beamwidth_deg(diameter_m, frequency_ghz):
    """Half-power beamwidth (degrees)."""
    wavelength = C / (frequency_ghz * 1e9)
    return np.degrees(70 * wavelength / diameter_m)

def noise_power_dbm(bandwidth_hz, noise_temp_k):
    """Noise power in dBm."""
    noise_w = K_BOLTZMANN * noise_temp_k * bandwidth_hz
    return 10 * np.log10(noise_w) + 30  # Convert to dBm

def rf_isl_link_budget(tx_power_dbm, tx_ant_m, rx_ant_m, 
                       freq_ghz, range_km, bandwidth_mhz,
                       tx_loss_db=1, rx_loss_db=1, rx_temp_k=500):
    """Complete RF ISL link budget."""
    # Gains
    tx_gain = antenna_gain_db(tx_ant_m, freq_ghz)
    rx_gain = antenna_gain_db(rx_ant_m, freq_ghz)
    
    # EIRP
    eirp = tx_power_dbm + tx_gain - tx_loss_db
    
    # Path loss
    fspl = fspl_db(range_km, freq_ghz)
    
    # Received power
    rx_power = eirp - fspl + rx_gain - rx_loss_db
    
    # Noise
    bw_hz = bandwidth_mhz * 1e6
    noise = noise_power_dbm(bw_hz, rx_temp_k)
    
    # C/N
    cn_db = rx_power - noise
    
    # G/T
    gt = rx_gain - 10 * np.log10(rx_temp_k)
    
    # Beamwidth
    bw_tx = beamwidth_deg(tx_ant_m, freq_ghz)
    bw_rx = beamwidth_deg(rx_ant_m, freq_ghz)
    
    return {
        'tx_power_dbm': tx_power_dbm,
        'tx_gain_dbi': tx_gain,
        'rx_gain_dbi': rx_gain,
        'eirp_dbm': eirp,
        'fspl_db': fspl,
        'rx_power_dbm': rx_power,
        'noise_floor_dbm': noise,
        'cn_db': cn_db,
        'gt_dbk': gt,
        'tx_beamwidth_deg': bw_tx,
        'rx_beamwidth_deg': bw_rx,
        'pointing_req_deg': bw_tx / 10,
    }

def estimate_data_rate(cn_db, bandwidth_mhz, modulation='QPSK'):
    """Estimate achievable data rate."""
    spectral_eff = {
        'BPSK': 1,
        'QPSK': 2,
        '8PSK': 3,
        '16QAM': 4,
        '64QAM': 6,
    }
    cn_req = {
        'BPSK': 6,
        'QPSK': 10,
        '8PSK': 14,
        '16QAM': 17,
        '64QAM': 23,
    }
    
    if cn_db >= cn_req[modulation]:
        rate_mbps = bandwidth_mhz * spectral_eff[modulation] * 0.8  # 80% efficiency
        margin = cn_db - cn_req[modulation]
        return rate_mbps, margin
    return 0, cn_db - cn_req[modulation]

# Analysis
print("RF ISL Link Budget Analysis")
print("=" * 60)

# Scenario 1: LEO intra-plane ISL (Ka-band)
print("\n1. LEO Intra-Plane ISL (Ka-band, 2000 km)")
print("-" * 60)
budget1 = rf_isl_link_budget(
    tx_power_dbm=37,      # 5 W
    tx_ant_m=0.5,
    rx_ant_m=0.5,
    freq_ghz=25,
    range_km=2000,
    bandwidth_mhz=100,
)

for k, v in budget1.items():
    if 'db' in k.lower() or 'dbi' in k.lower() or 'dbk' in k.lower() or 'dbm' in k.lower():
        print(f"  {k:20}: {v:+.1f} dB")
    elif 'deg' in k.lower():
        print(f"  {k:20}: {v:.2f}°")
    else:
        print(f"  {k:20}: {v:.1f}")

rate, margin = estimate_data_rate(budget1['cn_db'], 100, '16QAM')
print(f"\n  Achievable rate (16QAM): {rate:.0f} Mbps, margin: {margin:.1f} dB")

# Scenario 2: LEO inter-plane ISL (longer range)
print("\n2. LEO Inter-Plane ISL (V-band, 5000 km)")
print("-" * 60)
budget2 = rf_isl_link_budget(
    tx_power_dbm=40,      # 10 W
    tx_ant_m=0.3,         # Smaller antenna for V-band
    rx_ant_m=0.3,
    freq_ghz=60,          # V-band
    range_km=5000,
    bandwidth_mhz=500,
)

for k, v in budget2.items():
    if 'db' in k.lower() or 'dbi' in k.lower() or 'dbk' in k.lower() or 'dbm' in k.lower():
        print(f"  {k:20}: {v:+.1f} dB")
    elif 'deg' in k.lower():
        print(f"  {k:20}: {v:.2f}°")
    else:
        print(f"  {k:20}: {v:.1f}")

rate2, margin2 = estimate_data_rate(budget2['cn_db'], 500, 'QPSK')
print(f"\n  Achievable rate (QPSK): {rate2:.0f} Mbps, margin: {margin2:.1f} dB")

# Comparison table
print("\n" + "=" * 60)
print("RF ISL Band Comparison (2000 km link)")
print("-" * 60)
bands = [
    ('Ku', 15, 0.6),
    ('Ka', 25, 0.5),
    ('V', 60, 0.3),
]

print(f"{'Band':6} {'Freq':>8} {'Dish':>8} {'FSPL':>10} {'Gain':>8} {'BW':>8}")
print(f"{'':6} {'(GHz)':>8} {'(m)':>8} {'(dB)':>10} {'(dBi)':>8} {'(deg)':>8}")
print("-" * 60)
for name, freq, dish in bands:
    fspl = fspl_db(2000, freq)
    gain = antenna_gain_db(dish, freq)
    bw = beamwidth_deg(dish, freq)
    print(f"{name:6} {freq:>8.0f} {dish:>8.1f} {fspl:>10.1f} {gain:>8.1f} {bw:>8.2f}")
```

<section class="lesson-pitfalls">

## Pitfalls

1. **Antenna pointing error.** Even 0.1° pointing error at Ka-band can lose 3 dB — critical for margin.

2. **Inter-plane seam problem.** At polar regions, adjacent planes diverge; inter-plane ISLs may need to hand off.

3. **Range variation.** Inter-plane ISL range can vary 2:1 over an orbit — link budget must handle worst case.

4. **Doppler shift.** LEO ISLs see ±50 kHz Doppler at Ku-band; receivers need tracking loops.

5. **Regulatory coordination.** ISL frequencies still need ITU coordination even in space.

6. **Eclipse thermal cycling.** RF electronics must survive -40°C to +80°C swings every orbit.

7. **Sun interference.** When Sun is in antenna boresight, noise temperature spikes dramatically.
