## Why this matters

Before any spacecraft approaches a crewed station or high-value asset, mission planners define invisible boundaries in space—safety zones that govern where visiting vehicles can and cannot go. These zones protect against collision, thruster plume damage, and communication interference. Understanding zone geometry is essential for flight software development, mission planning, and real-time operations.

## Apply

**Safety zone geometry calculations:**

```python
import numpy as np

def ellipsoid_boundary(semi_axes, num_points=100):
    """Generate points on an ellipsoid surface."""
    # Semi-axes: (a, b, c) for x, y, z
    a, b, c = semi_axes
    
    theta = np.linspace(0, 2*np.pi, num_points)
    phi = np.linspace(0, np.pi, num_points//2)
    
    theta, phi = np.meshgrid(theta, phi)
    
    x = a * np.sin(phi) * np.cos(theta)
    y = b * np.sin(phi) * np.sin(theta)
    z = c * np.cos(phi)
    
    return x, y, z

def point_in_ellipsoid(point, semi_axes):
    """Check if point is inside ellipsoid."""
    x, y, z = point
    a, b, c = semi_axes
    return (x/a)**2 + (y/b)**2 + (z/c)**2 <= 1

def point_in_corridor(point, half_angle_deg, corridor_axis='x'):
    """Check if point is within approach corridor."""
    x, y, z = point
    half_angle = np.radians(half_angle_deg)
    
    if corridor_axis == 'x':  # V-bar approach
        if x >= 0:  # Must be approaching (positive x toward target)
            return False
        lateral_dist = np.sqrt(y**2 + z**2)
        axial_dist = abs(x)
        return lateral_dist <= axial_dist * np.tan(half_angle)
    return False

def check_zone(position_m, zones):
    """Determine which zone a position is in."""
    for zone_name, zone_def in sorted(zones.items(), 
                                       key=lambda x: x[1].get('priority', 0)):
        if zone_def['type'] == 'ellipsoid':
            if point_in_ellipsoid(position_m, zone_def['semi_axes']):
                return zone_name
        elif zone_def['type'] == 'sphere':
            if np.linalg.norm(position_m) <= zone_def['radius']:
                return zone_name
        elif zone_def['type'] == 'corridor':
            if point_in_corridor(position_m, zone_def['half_angle']):
                return zone_name
    return 'outside'

def gate_criteria(from_zone, to_zone, state):
    """Check if gate criteria are met for zone transition."""
    range_m = state['range_m']
    range_rate = state['range_rate_m_s']
    lateral_rate = state['lateral_rate_m_s']
    
    criteria = {
        ('zone4', 'zone3'): {
            'range_rate_max': 10.0,
            'lateral_rate_max': 2.0,
            'nav_converged': True,
            'comm_verified': True,
        },
        ('zone3', 'zone2'): {
            'range_rate_max': 2.0,
            'lateral_rate_max': 0.5,
            'corridor_offset_max': 50.0,
            'cam_fuel_min': 10.0,
        },
        ('zone2', 'zone1'): {
            'range_rate_max': 0.5,
            'lateral_rate_max': 0.1,
            'alignment_max_deg': 3.0,
            'lidar_locked': True,
        },
    }
    
    key = (from_zone, to_zone)
    if key not in criteria:
        return True, []
    
    c = criteria[key]
    failures = []
    
    if range_rate > c.get('range_rate_max', float('inf')):
        failures.append(f"range_rate {range_rate:.1f} > {c['range_rate_max']}")
    if lateral_rate > c.get('lateral_rate_max', float('inf')):
        failures.append(f"lateral_rate {lateral_rate:.2f} > {c['lateral_rate_max']}")
    
    return len(failures) == 0, failures

# ISS zone definitions
ISS_ZONES = {
    'zone1': {'type': 'corridor', 'half_angle': 10, 'priority': 1},
    'zone2': {'type': 'sphere', 'radius': 200, 'priority': 2},
    'zone3': {'type': 'ellipsoid', 'semi_axes': (2000, 1500, 1500), 'priority': 3},
    'zone4': {'type': 'ellipsoid', 'semi_axes': (4000, 3000, 3000), 'priority': 4},
}

# Analysis
print("Safety Zone Analysis")
print("=" * 60)

# Test positions
test_positions = [
    ([-5000, 0, 0], "5 km on V-bar"),
    ([-3000, 500, 0], "3 km, 500m lateral offset"),
    ([-1500, 0, 0], "1.5 km on V-bar"),
    ([-150, 0, 0], "150 m on V-bar"),
    ([-100, 50, 0], "100 m, 50m offset"),
]

print("\nPosition Zone Classification:")
print("-" * 60)
for pos, desc in test_positions:
    zone = check_zone(pos, ISS_ZONES)
    print(f"  {desc:30} → {zone}")

# Gate transition check
print("\n" + "-" * 60)
print("Gate Transition Analysis:")

states = [
    {'range_m': 4000, 'range_rate_m_s': 8.0, 'lateral_rate_m_s': 1.0},
    {'range_m': 4000, 'range_rate_m_s': 15.0, 'lateral_rate_m_s': 1.0},
    {'range_m': 2000, 'range_rate_m_s': 1.5, 'lateral_rate_m_s': 0.3},
    {'range_m': 200, 'range_rate_m_s': 0.3, 'lateral_rate_m_s': 0.05},
]

transitions = [
    ('zone4', 'zone3'),
    ('zone4', 'zone3'),
    ('zone3', 'zone2'),
    ('zone2', 'zone1'),
]

print(f"\n  {'Transition':20} {'Range':>8} {'Rng Rate':>10} {'Result':>10}")
print(f"  {'-'*20} {'-'*8} {'-'*10} {'-'*10}")
for (from_z, to_z), state in zip(transitions, states):
    passed, failures = gate_criteria(from_z, to_z, state)
    result = "PASS" if passed else "FAIL"
    print(f"  {from_z}→{to_z:12} {state['range_m']:>7}m "
          f"{state['range_rate_m_s']:>8.1f}m/s {result:>10}")
    if failures:
        for f in failures:
            print(f"    ⚠ {f}")

# Plume exclusion
print("\n" + "-" * 60)
print("Plume Exclusion Zones:")
exclusion_cones = [
    {'name': 'Solar Array', 'half_angle': 30, 'min_range': 50},
    {'name': 'Radiator', 'half_angle': 20, 'min_range': 30},
    {'name': 'Antenna', 'half_angle': 15, 'min_range': 100},
]

for cone in exclusion_cones:
    solid_angle = 2 * np.pi * (1 - np.cos(np.radians(cone['half_angle'])))
    fraction = solid_angle / (4 * np.pi) * 100
    print(f"  {cone['name']:15} ±{cone['half_angle']}° "
          f"({fraction:.1f}% of sphere), min {cone['min_range']}m")
```

<section class="lesson-pitfalls">

## Pitfalls

1. **Ellipsoid vs sphere confusion.** Far-field zones are often ellipsoids aligned with velocity vector, not spheres.

2. **Corridor half-angle varies.** Different docking ports may have different corridor geometries.

3. **Gate criteria are AND conditions.** All criteria must pass; one failure blocks transition.

4. **Zone boundaries are dynamic.** During ISS reboost or attitude changes, zones may temporarily expand.

5. **Plume zones depend on thruster selection.** Which thrusters fire determines which exclusions apply.

6. **RF zones are bidirectional.** Both visiting vehicle and station transmitters create exclusion zones.

7. **Lighting affects zone ops.** Some optical sensors require daylight; zone transitions may be lighting-constrained.
