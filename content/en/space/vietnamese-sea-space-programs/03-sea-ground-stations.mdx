## Ground Station Link Analysis

### LEO Pass Geometry Calculation

```python
import math
import numpy as np

def leo_pass_analysis(station_lat, station_lon, sat_alt_km, inclination_deg, min_elevation=5):
    """
    Analyze LEO satellite pass geometry for a ground station.
    
    Args:
        station_lat: Station latitude in degrees
        station_lon: Station longitude in degrees  
        sat_alt_km: Satellite altitude in km
        inclination_deg: Orbital inclination in degrees
        min_elevation: Minimum elevation angle in degrees
    
    Returns:
        Pass statistics dictionary
    """
    R_earth = 6371  # km
    
    # Maximum slant range (at minimum elevation)
    min_el_rad = math.radians(min_elevation)
    
    # Using law of cosines for slant range
    # d = R*sin(el) + sqrt((R*sin(el))^2 + h^2 + 2*R*h)
    # Simplified for low elevation:
    cos_gamma = math.cos(math.pi/2 - min_el_rad)
    max_slant_range = math.sqrt(R_earth**2 + (R_earth + sat_alt_km)**2 - 
                                 2*R_earth*(R_earth + sat_alt_km)*cos_gamma)
    
    # Minimum slant range (at zenith)
    min_slant_range = sat_alt_km
    
    # Visibility circle radius (half-angle from sub-satellite point)
    # sin(rho) = R_earth * cos(el_min) / (R_earth + h)
    rho_rad = math.acos(R_earth * math.cos(min_el_rad) / (R_earth + sat_alt_km))
    visibility_radius_km = rho_rad * R_earth
    
    # Pass duration estimate (overhead pass)
    orbital_velocity = math.sqrt(398600.4 / (R_earth + sat_alt_km))  # km/s
    ground_track_arc = 2 * visibility_radius_km
    max_pass_duration_min = ground_track_arc / orbital_velocity / 60
    
    # Number of passes per day (approximate)
    orbital_period_min = 2 * math.pi * math.sqrt((R_earth + sat_alt_km)**3 / 398600.4) / 60
    orbits_per_day = 1440 / orbital_period_min
    
    # Passes visible depend on station latitude vs inclination
    if abs(station_lat) <= inclination_deg:
        # Station within coverage band - both ascending and descending
        visible_fraction = 0.2  # Approximate
    else:
        visible_fraction = 0.1
    
    passes_per_day = orbits_per_day * visible_fraction
    
    results = {
        "station": f"{station_lat:.2f}°N, {station_lon:.2f}°E",
        "satellite_altitude_km": sat_alt_km,
        "min_slant_range_km": min_slant_range,
        "max_slant_range_km": max_slant_range,
        "visibility_radius_km": visibility_radius_km,
        "max_pass_duration_min": max_pass_duration_min,
        "orbital_period_min": orbital_period_min,
        "orbits_per_day": orbits_per_day,
        "estimated_passes_per_day": passes_per_day
    }
    
    return results

def print_pass_analysis():
    """
    Analyze MicroDragon passes over Hoa Lac station.
    """
    # Hoa Lac coordinates
    hoa_lac_lat = 21.01
    hoa_lac_lon = 105.52
    
    # MicroDragon orbit
    microdragon_alt = 511
    microdragon_inc = 97.31
    
    results = leo_pass_analysis(
        hoa_lac_lat, hoa_lac_lon, 
        microdragon_alt, microdragon_inc,
        min_elevation=5
    )
    
    print("=== MicroDragon Pass Analysis: Hoa Lac Station ===")
    print(f"Station:                 {results['station']}")
    print(f"Satellite altitude:      {results['satellite_altitude_km']:.0f} km")
    print(f"Min slant range:         {results['min_slant_range_km']:.0f} km (at zenith)")
    print(f"Max slant range:         {results['max_slant_range_km']:.0f} km (at 5° elevation)")
    print(f"Visibility radius:       {results['visibility_radius_km']:.0f} km")
    print(f"Max pass duration:       {results['max_pass_duration_min']:.1f} minutes")
    print(f"Orbital period:          {results['orbital_period_min']:.1f} minutes")
    print(f"Orbits per day:          {results['orbits_per_day']:.1f}")
    print(f"Est. passes/day:         {results['estimated_passes_per_day']:.1f}")
    
    # Data volume estimate
    data_rate_mbps = 40  # X-band
    avg_pass_min = results['max_pass_duration_min'] * 0.7  # Not all overhead
    data_per_pass_gb = data_rate_mbps * avg_pass_min * 60 / 8 / 1000
    daily_data_gb = data_per_pass_gb * results['estimated_passes_per_day']
    
    print(f"")
    print(f"=== Data Volume Estimates ===")
    print(f"X-band data rate:        {data_rate_mbps} Mbps")
    print(f"Avg effective pass:      {avg_pass_min:.1f} minutes")
    print(f"Data per pass:           {data_per_pass_gb:.1f} GB")
    print(f"Daily data volume:       {daily_data_gb:.1f} GB")

if __name__ == "__main__":
    print_pass_analysis()
```

### G/T and Link Margin

```python
import math

def ground_station_link_budget(
    antenna_diameter_m,
    antenna_efficiency,
    system_temp_k,
    frequency_ghz,
    satellite_eirp_dbw,
    slant_range_km
):
    """
    Calculate ground station receive link budget.
    """
    c = 3e8
    wavelength_m = c / (frequency_ghz * 1e9)
    
    # Antenna gain
    aperture = math.pi * (antenna_diameter_m / 2) ** 2
    effective_aperture = antenna_efficiency * aperture
    gain_linear = (4 * math.pi * effective_aperture) / wavelength_m**2
    gain_db = 10 * math.log10(gain_linear)
    
    # G/T
    noise_temp_db = 10 * math.log10(system_temp_k)
    g_over_t = gain_db - noise_temp_db
    
    # Free space path loss
    fspl_db = 20 * math.log10(4 * math.pi * slant_range_km * 1e3 / wavelength_m)
    
    # C/N0
    k_db = -228.6  # dB(W/Hz/K)
    c_over_n0 = satellite_eirp_dbw - fspl_db + g_over_t - k_db
    
    return {
        "antenna_gain_db": gain_db,
        "g_over_t_db_k": g_over_t,
        "fspl_db": fspl_db,
        "c_over_n0_db_hz": c_over_n0
    }

# Hoa Lac primary antenna receiving MicroDragon
results = ground_station_link_budget(
    antenna_diameter_m=7.3,
    antenna_efficiency=0.55,
    system_temp_k=120,  # X-band system temp
    frequency_ghz=8.2,  # X-band
    satellite_eirp_dbw=10,  # Typical LEO EO satellite
    slant_range_km=800  # Mid-pass
)

print("=== Hoa Lac X-band Link Budget ===")
for key, val in results.items():
    print(f"{key}: {val:.2f}")
```

## Common Pitfalls

### 1. Horizon Masking
Mountain terrain around Hoa Lac creates 3-5° horizon mask in some directions, reducing usable pass time by 10-20%.

### 2. Interference in Urban Areas
Singapore teleport faces RF interference challenges. Ku-band uplinks near 14 GHz can conflict with 5G deployments (3.3-3.8 GHz harmonics).

### 3. Tropical Weather Effects
- Rain attenuation severe above 10 GHz
- Lightning protection critical
- Humidity affects waveguide losses
- Monsoon wind loads on antennas

### 4. Timing Synchronization
LEO tracking requires &lt;1ms timing accuracy. GPS-disciplined oscillators standard, but GPS jamming/spoofing increasingly problematic.

### 5. Regulatory Coordination
Each country has different spectrum allocation. Cross-border data sharing requires bilateral agreements beyond technical capability.
