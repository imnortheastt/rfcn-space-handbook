## Why this matters

GEO cannot see polar regions. Russia solved this with Molniya orbits—highly elliptical paths that "hang" over high latitudes for hours. Meanwhile, Earth observation satellites need consistent lighting conditions, achieved through Sun-synchronous orbits (SSO). These specialized orbits fill gaps that circular equatorial orbits cannot.

### SSO precession math

```

┌─────────────────────────────────────────────────────────────────────┐

│  ┌───────────────────────────────────────────────────────┐         │
│  │                                                        │         │
│  │  Ω̇ = -3/2 × J2 × (Re/a)² × n × cos(i) / (1-e²)²     │         │
│  │                                                        │         │
│  │  Where:                                               │         │
│  │  • J2 = 1.08263×10⁻³ (Earth oblateness)              │         │
```

## Apply

**Molniya and SSO orbital calculations:**

```python
import numpy as np

# Constants
MU_EARTH = 398600.4418  # km³/s²
R_EARTH = 6378.137  # km (equatorial)
J2 = 1.08263e-3
EARTH_ROTATION_DEG_DAY = 360.9856  # sidereal
SUN_MOTION_DEG_DAY = 0.9856  # Earth around Sun

def molniya_parameters():
    """Calculate Molniya orbit parameters."""
    # 12-hour period
    period_s = 12 * 3600
    
    # Semi-major axis from Kepler's third law
    a = (MU_EARTH * (period_s / (2 * np.pi))**2) ** (1/3)
    
    # Critical inclination (no apsidal rotation)
    i_critical = np.degrees(np.arcsin(np.sqrt(0.8)))
    
    # Typical Molniya parameters
    perigee = 500  # km
    r_perigee = R_EARTH + perigee
    
    # Eccentricity from a and r_perigee
    # r_perigee = a(1-e) → e = 1 - r_perigee/a
    e = 1 - r_perigee / a
    
    # Apogee
    r_apogee = a * (1 + e)
    apogee = r_apogee - R_EARTH
    
    # Velocities
    v_perigee = np.sqrt(MU_EARTH * (2/r_perigee - 1/a))
    v_apogee = np.sqrt(MU_EARTH * (2/r_apogee - 1/a))
    
    return {
        'period_hours': 12,
        'semi_major_axis_km': a,
        'eccentricity': e,
        'inclination_deg': i_critical,
        'perigee_km': perigee,
        'apogee_km': apogee,
        'v_perigee_km_s': v_perigee,
        'v_apogee_km_s': v_apogee,
        'dwell_time_hours': 8  # approximate above 60°N
    }

def sso_inclination(altitude_km):
    """Calculate required inclination for sun-synchronous orbit."""
    a = R_EARTH + altitude_km
    
    # Mean motion (rad/s)
    n = np.sqrt(MU_EARTH / a**3)
    
    # Required RAAN rate (rad/s)
    omega_dot_required = np.radians(SUN_MOTION_DEG_DAY) / 86400
    
    # From RAAN precession formula (circular orbit, e=0):
    # Ω̇ = -3/2 × J2 × (Re/a)² × n × cos(i)
    # Solve for cos(i)
    cos_i = -omega_dot_required / (1.5 * J2 * (R_EARTH/a)**2 * n)
    
    if abs(cos_i) > 1:
        return None  # No SSO solution at this altitude
    
    i = np.degrees(np.arccos(cos_i))
    
    # Period
    period_min = 2 * np.pi / n / 60
    
    return {
        'altitude_km': altitude_km,
        'inclination_deg': i,
        'period_min': period_min,
        'is_retrograde': i > 90
    }

def sso_local_time_drift(altitude_km, inclination_error_deg):
    """Calculate local time drift from inclination error."""
    # Nominal SSO
    nominal = sso_inclination(altitude_km)
    if nominal is None:
        return None
    
    a = R_EARTH + altitude_km
    n = np.sqrt(MU_EARTH / a**3)
    
    # Actual RAAN rate with error
    i_actual = np.radians(nominal['inclination_deg'] + inclination_error_deg)
    omega_dot_actual = -1.5 * J2 * (R_EARTH/a)**2 * n * np.cos(i_actual)
    
    # Required rate
    omega_dot_required = np.radians(SUN_MOTION_DEG_DAY) / 86400
    
    # Drift
    drift_rad_s = omega_dot_actual - omega_dot_required
    drift_deg_day = np.degrees(drift_rad_s) * 86400
    
    # Local time drift (360° = 24 hours)
    local_time_drift_min_day = drift_deg_day / 360 * 24 * 60
    
    return {
        'inclination_error_deg': inclination_error_deg,
        'raan_drift_deg_day': drift_deg_day,
        'local_time_drift_min_day': local_time_drift_min_day
    }

# Calculate and display
print("Molniya Orbit Analysis")
print("=" * 60)

mol = molniya_parameters()
print(f"\nMolniya Orbit Parameters:")
print(f"  Period: {mol['period_hours']} hours")
print(f"  Semi-major axis: {mol['semi_major_axis_km']:.0f} km")
print(f"  Eccentricity: {mol['eccentricity']:.3f}")
print(f"  Critical inclination: {mol['inclination_deg']:.2f}°")
print(f"  Perigee: {mol['perigee_km']:.0f} km")
print(f"  Apogee: {mol['apogee_km']:.0f} km")
print(f"  Velocity at perigee: {mol['v_perigee_km_s']:.2f} km/s")
print(f"  Velocity at apogee: {mol['v_apogee_km_s']:.2f} km/s")
print(f"  Dwell time above 60°N: ~{mol['dwell_time_hours']} hours")

print("\n" + "=" * 60)
print("Sun-Synchronous Orbit Analysis")
print("-" * 60)

for alt in [400, 600, 800, 1000, 1200]:
    sso = sso_inclination(alt)
    if sso:
        print(f"  {alt:4d} km → i = {sso['inclination_deg']:.2f}°, "
              f"T = {sso['period_min']:.1f} min "
              f"{'(retrograde)' if sso['is_retrograde'] else ''}")
    else:
        print(f"  {alt:4d} km → No SSO solution")

# Drift analysis
print("\n" + "-" * 60)
print("SSO Local Time Drift (at 700 km, per 0.1° inclination error):")
drift = sso_local_time_drift(700, 0.1)
if drift:
    print(f"  RAAN drift: {drift['raan_drift_deg_day']:.4f}°/day")
    print(f"  Local time drift: {drift['local_time_drift_min_day']:.2f} min/day")
    print(f"  Annual drift: {drift['local_time_drift_min_day'] * 365:.0f} min/year")
```

<section class="lesson-pitfalls">

## Pitfalls

1. **Critical inclination is not optional.** At any other angle, Molniya apogee drifts—orbit becomes useless within months.

2. **Molniya perigee is in Van Allen belts.** Satellites experience high radiation twice per orbit; limits component lifetime.

3. **SSO only works below ~1500 km.** Above this, J2 effect is too weak to achieve required precession rate.

4. **SSO inclination must be maintained.** 0.1° error causes ~10 minutes local time drift per year—needs station-keeping.

5. **Retrograde orbits are harder to launch.** SSO requires launching against Earth's rotation—loses ~0.4 km/s.

6. **12-hour Molniya has ground track repeating daily.** Handover between satellites happens at same time each day.

7. **SSO doesn't mean always-lit.** Dawn-dusk SSO always flies along terminator—useful for SAR, not optical imaging.
