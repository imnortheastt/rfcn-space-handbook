## Why this matters

Medium Earth Orbit (MEO) is the "Goldilocks zone" for navigation and some telecom: higher than LEO for better coverage, lower than GEO for acceptable latency. GPS, Galileo, GLONASS, and O3b mPOWER all operate in MEO. Understanding MEO physics explains why navigation constellations need only ~30 satellites while LEO broadband needs thousands.

## Apply

**MEO orbital calculations:**

```python
import numpy as np

# Constants
MU_EARTH = 398600.4418  # km³/s²
R_EARTH = 6371  # km
C = 299792.458  # km/s

def meo_parameters(altitude_km):
    """Calculate MEO orbital parameters."""
    a = R_EARTH + altitude_km
    
    # Orbital period
    period_s = 2 * np.pi * np.sqrt(a**3 / MU_EARTH)
    period_hr = period_s / 3600
    
    # Velocity
    velocity = np.sqrt(MU_EARTH / a)
    
    # Coverage (nadir angle for 5° min elevation)
    min_elev = 5
    rho = np.arcsin(R_EARTH * np.cos(np.radians(min_elev)) / a)
    nadir_angle = 90 - min_elev - np.degrees(rho)
    coverage_diameter = 2 * R_EARTH * np.radians(nadir_angle)
    
    return {
        'altitude_km': altitude_km,
        'period_hours': period_hr,
        'velocity_km_s': velocity,
        'coverage_diameter_km': coverage_diameter
    }

def gnss_visibility(altitude_km, min_elevation_deg=5):
    """
    Calculate number of GNSS satellites visible from a point.
    Simplified geometric model.
    """
    a = R_EARTH + altitude_km
    
    # Earth central angle visible from satellite
    rho = np.arcsin(R_EARTH * np.cos(np.radians(min_elevation_deg)) / a)
    earth_angle = np.pi/2 - np.radians(min_elevation_deg) - rho
    
    # Fraction of Earth surface visible
    coverage_fraction = (1 - np.cos(earth_angle)) / 2
    
    # For 24-satellite constellation evenly distributed
    # Expected visible satellites ≈ 24 × coverage_fraction × geometry_factor
    # Geometry factor accounts for non-uniform distribution
    geometry_factor = 1.5  # empirical
    expected_visible = 24 * coverage_fraction * geometry_factor
    
    return {
        'coverage_fraction': coverage_fraction,
        'earth_angle_deg': np.degrees(earth_angle),
        'expected_visible_sats': expected_visible
    }

def meo_latency(altitude_km, elevation_deg=30):
    """Calculate MEO link latency."""
    h = altitude_km
    eps = np.radians(elevation_deg)
    
    # Slant range
    rho = np.arcsin(R_EARTH * np.cos(eps) / (R_EARTH + h))
    slant_range = (R_EARTH + h) * np.sin(np.pi/2 - eps - rho) / np.cos(eps)
    
    # Latency
    one_way_ms = slant_range / C * 1000
    
    return {
        'slant_range_km': slant_range,
        'one_way_ms': one_way_ms,
        'round_trip_ms': 2 * one_way_ms
    }

# Compare MEO systems
print("MEO System Comparison")
print("=" * 65)

systems = [
    ("O3b mPOWER", 8062),
    ("GLONASS", 19130),
    ("GPS", 20180),
    ("Galileo", 23222),
]

for name, alt in systems:
    params = meo_parameters(alt)
    latency = meo_latency(alt, 30)
    
    print(f"\n{name} ({alt} km):")
    print(f"  Period: {params['period_hours']:.2f} hours")
    print(f"  Velocity: {params['velocity_km_s']:.2f} km/s")
    print(f"  Coverage diameter: {params['coverage_diameter_km']:.0f} km")
    print(f"  RTT (30° elev): {latency['round_trip_ms']:.1f} ms")

# GPS visibility analysis
print("\n" + "=" * 65)
print("GPS Visibility Analysis:")
print("-" * 65)

vis = gnss_visibility(20180, 5)
print(f"  Coverage fraction per satellite: {vis['coverage_fraction']*100:.1f}%")
print(f"  Earth angle visible: {vis['earth_angle_deg']:.1f}°")
print(f"  Expected visible satellites: {vis['expected_visible_sats']:.1f}")
print(f"  (Need 4 for position fix, typically see 6-12)")

# Constellation sizing comparison
print("\n" + "=" * 65)
print("Constellation Size Comparison:")
print("-" * 65)
print(f"  {'System':<20} {'Altitude':<12} {'Satellites':<12} {'Use Case'}")
print(f"  {'-'*20} {'-'*12} {'-'*12} {'-'*20}")
data = [
    ("GPS", "20,180 km", "31", "Navigation"),
    ("Galileo", "23,222 km", "30", "Navigation"),
    ("O3b mPOWER", "8,062 km", "11", "Telecom"),
    ("Starlink", "550 km", "4,000+", "Broadband"),
]
for row in data:
    print(f"  {row[0]:<20} {row[1]:<12} {row[2]:<12} {row[3]}")
```

<section class="lesson-pitfalls">

## Pitfalls

1. **Van Allen exposure.** MEO satellites need radiation hardening; adds mass and cost.

2. **Slot region myth.** O3b's 8,000 km is lower radiation but still requires protection.

3. **Equatorial limitation.** O3b can't serve polar regions; not true global coverage.

4. **12-hour resonance.** GPS's 11:58 period chosen specifically for repeating ground track.

5. **Tracking complexity.** Unlike GEO, MEO needs tracking antennas; steerable or phased array.

6. **Debris persistence.** MEO debris doesn't naturally decay; stays for millennia.

7. **Launch cost.** Higher altitude = more delta-v = more expensive to reach than LEO.
