## Why this matters

Low Earth Orbit (LEO) is where the action is: Starlink, OneWeb, Kuiper, and most Earth observation satellites operate here. LEO offers low latency (25-50 ms round-trip) and manageable link budgets, but demands large constellations for global coverage because each satellite sees only a small slice of Earth. Understanding LEO physics is foundational to satellite communications engineering.

## Apply

**LEO orbital calculations:**

```python
import numpy as np

# Constants
MU_EARTH = 398600.4418  # km³/s² - Earth's gravitational parameter
R_EARTH = 6371  # km - Earth's radius
C = 299792.458  # km/s - speed of light

def leo_orbital_parameters(altitude_km):
    """
    Calculate LEO orbital parameters.
    """
    a = R_EARTH + altitude_km  # semi-major axis
    
    # Orbital period (Kepler's third law)
    period_s = 2 * np.pi * np.sqrt(a**3 / MU_EARTH)
    period_min = period_s / 60
    
    # Orbital velocity
    velocity_km_s = np.sqrt(MU_EARTH / a)
    
    # Ground track velocity (projection on Earth surface)
    ground_velocity = velocity_km_s * R_EARTH / a
    
    return {
        'altitude_km': altitude_km,
        'semi_major_axis_km': a,
        'period_min': period_min,
        'orbits_per_day': 1440 / period_min,
        'velocity_km_s': velocity_km_s,
        'ground_velocity_km_s': ground_velocity
    }

def coverage_diameter(altitude_km, min_elevation_deg):
    """
    Calculate coverage diameter on Earth surface.
    """
    h = altitude_km
    eps = np.radians(min_elevation_deg)
    
    # Central angle from satellite to coverage edge
    # Using spherical geometry
    rho = np.arcsin(R_EARTH * np.cos(eps) / (R_EARTH + h))
    lambda_0 = np.pi/2 - eps - rho
    
    # Coverage radius and diameter
    coverage_radius_km = R_EARTH * lambda_0
    coverage_diameter_km = 2 * coverage_radius_km
    
    # Coverage area
    coverage_area_km2 = 2 * np.pi * R_EARTH**2 * (1 - np.cos(lambda_0))
    
    return {
        'coverage_diameter_km': coverage_diameter_km,
        'coverage_area_km2': coverage_area_km2,
        'central_angle_deg': np.degrees(lambda_0)
    }

def link_latency(altitude_km, elevation_deg):
    """
    Calculate one-way space segment latency.
    """
    h = altitude_km
    eps = np.radians(elevation_deg)
    
    # Slant range calculation
    rho = np.arcsin(R_EARTH * np.cos(eps) / (R_EARTH + h))
    slant_range = (R_EARTH + h) * np.sin(np.pi/2 - eps - rho) / np.cos(eps)
    
    # One-way delay
    delay_ms = slant_range / C * 1000
    
    return {
        'slant_range_km': slant_range,
        'one_way_delay_ms': delay_ms,
        'round_trip_ms': 2 * delay_ms
    }

def min_constellation_size(altitude_km, min_elevation_deg, overlap_factor=1.5):
    """
    Estimate minimum constellation size for global coverage.
    """
    earth_surface_km2 = 4 * np.pi * R_EARTH**2
    cov = coverage_diameter(altitude_km, min_elevation_deg)
    
    # Minimum satellites (no overlap)
    min_sats = earth_surface_km2 / cov['coverage_area_km2']
    
    # Practical with overlap
    practical_sats = min_sats * overlap_factor
    
    return {
        'earth_surface_km2': earth_surface_km2,
        'coverage_per_sat_km2': cov['coverage_area_km2'],
        'theoretical_minimum': int(np.ceil(min_sats)),
        'practical_estimate': int(np.ceil(practical_sats))
    }

# Example: Compare LEO altitudes
print("LEO Orbital Comparison")
print("=" * 60)

altitudes = [400, 550, 1200]
for alt in altitudes:
    params = leo_orbital_parameters(alt)
    cov = coverage_diameter(alt, 25)
    lat = link_latency(alt, 45)
    
    print(f"\nAltitude: {alt} km")
    print(f"  Period: {params['period_min']:.1f} min ({params['orbits_per_day']:.1f} orbits/day)")
    print(f"  Velocity: {params['velocity_km_s']:.2f} km/s")
    print(f"  Coverage diameter (25° elev): {cov['coverage_diameter_km']:.0f} km")
    print(f"  RTT (45° elev): {lat['round_trip_ms']:.1f} ms")

# Constellation sizing
print("\n" + "=" * 60)
print("Constellation Sizing (25° min elevation):")
print("-" * 60)

for alt in [550, 1200]:
    size = min_constellation_size(alt, 25)
    print(f"\n{alt} km altitude:")
    print(f"  Theoretical minimum: {size['theoretical_minimum']} satellites")
    print(f"  Practical (1.5× overlap): {size['practical_estimate']} satellites")

# Real constellation comparison
print("\n" + "=" * 60)
print("Real LEO Constellations:")
print("-" * 60)
constellations = [
    ("Starlink Gen1", 550, 53, 1584),
    ("Starlink Gen2", 340, 43, 7500),
    ("OneWeb", 1200, 87.9, 648),
    ("Kuiper", 630, 51.9, 3236),
]
for name, alt, inc, sats in constellations:
    print(f"  {name:15} {alt:4} km, {inc:5.1f}°, {sats:5} sats")
```

## Pitfalls

1. **Van Allen belts.** Avoid 1,000-2,000 km for crewed missions; radiation peaks there.

2. **Orbital debris.** Popular altitudes (500-600 km) are crowded; collision risk increasing.

3. **Solar activity.** Atmospheric drag increases during solar maximum; satellites decay faster.

4. **Sun synchronous vs inclined.** SSO (98°) for Earth observation; lower inclination for telecom.

5. **Eclipses.** LEO satellites enter Earth's shadow every orbit; battery thermal cycling.

6. **Doppler tracking.** User terminals must handle ±50 kHz shifts; complicates modulation.

7. **Handover rate.** Every 60-90 seconds for LEO vs never for GEO; protocol complexity.
